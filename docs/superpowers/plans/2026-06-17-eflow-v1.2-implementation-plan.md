**Goal:** 在 v1.1 基础上完成 (1) P1 债务清理（L2 cache key 规范化 + 步进回填 + Concierge 死代码清理）；(2) Orchestrator 真正并行派发（step_to_layer + SubagentHandle 暴露 Subagent 引用 + Pool timeout-based idle cleanup）；(3) TUI 交互层（ratatui + crossterm 替换 main.rs 行输入模式）。

```rust
/// v1.2 D1: 把 capability 三处内联 CacheKey 构造抽到一处。
///
/// 设计选择（v1.1 plan §B1 + 跨阶段 D4 回顾）：
/// - `task_signature` **只含** step.action + step.tool，不含 params
///   （params 含时间戳/路径/任何字节 → cache miss；user 体验是「为什么我第二次没命中」）
/// - `intent_type` 由调用方按角色（Decisioner/Executor/Feedbacker）传入
/// - `model` 留空：Router 在 chat_cached 入口再注入
/// - `risk_level` / `profile_name` / 两个 bucket 默认 0
#[must_use]
pub fn cache_key_for_step(
    step: &crate::common::types::TaskStep,
    intent_type: crate::common::types::IntentType,
    risk_level: crate::common::types::RiskLevel,
    profile_name: &str,
) -> CacheKey {
    CacheKey {
        intent_type,
        task_signature: format!(
            "{}:{}:{}",
            intent_label(intent_type),
            step.tool,
            step.action
        ),
        context_profile: ContextProfile {
            conversation_depth_bucket: 0,
            file_count_bucket: 0,
            risk_level,
            profile_name: profile_name.to_string(),
        },
        model: String::new(),
    }
}

fn intent_label(it: crate::common::types::IntentType) -> &'static str {
    use crate::common::types::IntentType;
    match it {
        IntentType::CodeReview => "code_review",
        IntentType::BugFix => "bug_fix",
        IntentType::DataQuery => "data_query",
        IntentType::FileRead => "file_read",
        IntentType::FileWrite => "file_write",
        IntentType::CommandExecute => "command_execute",
        IntentType::WebFetch => "web_fetch",
        IntentType::Chat => "chat",
        IntentType::Unknown => "unknown",
    }
}
```

pub struct SubagentHandle {
    id: Uuid,
    pool_active: Arc<std::sync::Mutex<HashMap<Uuid, Subagent>>>,
}

impl SubagentHandle {
    #[must_use]
    pub fn id(&self) -> Uuid {
        self.id
    }

    /// v1.2 E1: 暴露 &Subagent，让 Orchestrator 拿到 agent 角色 / 能力做调度决策
    /// 注意：返回的引用受 handle 的 drop 语义保护，handle 还活着 → active map 还在
    #[must_use]
    pub fn subagent(&self) -> Option<std::cell::Ref<'_, Subagent>> {
        // 简化版：用 OnceCell 或 cell::Ref 包装 active map entry
        // ——但 std::sync::Mutex 不支持 cell::Ref。改用 plain &Subagent via map guard
        // （handle drop → map.remove(self.id)，所以 &Subagent 在 handle alive 期间安全）
        let map = self.pool_active.lock().ok()?;
        map.get(&self.id)
    }
}

impl Drop for SubagentHandle {
    fn drop(&mut self) {
        if let Ok(mut map) = self.pool_active.lock() {
            map.remove(&self.id);
        }
    }
}
```

> **简化说明**：`pool_active` 用 `std::sync::Mutex`，直接 lock 后返回 `&Subagent` 在 handle alive 期间是安全的（drop 才会 remove）。如果想更严格地表达「借给 handle 用」，可以把返回类型包成 `MutexGuard<'_, HashMap>`，但这要求 caller 持有 guard 跨 await，对 Orchestrator 不方便。**当前方案**接受 borrow 期间假设 handle 不会 drop（Orchestrator 内部逻辑保证）。

impl Orchestrator {
    // ...

    /// v1.2 E3: 把 TaskPlan 按依赖分层，每层内的步骤可并行执行。
    ///
    /// 算法：广度优先遍历 depends_on 图。
    /// - 无依赖（depends_on=None）→ layer 0
    /// - 依赖的步骤在 layer N → 本步骤在 layer N+1
    /// - 多依赖取最大 layer + 1
    ///
    /// 返回 Vec<Vec<u8>>，外层下标 = layer 索引，内层 = 该层步骤的 order 列表
    pub(crate) fn compute_step_layers(plan: &TaskPlan) -> Vec<Vec<u8>> {
        let mut step_to_layer: std::collections::HashMap<u8, usize> =
            std::collections::HashMap::new();
        let mut layers: Vec<Vec<u8>> = vec![vec![]];

        for step in &plan.steps {
            let layer = match step.depends_on {
                None => 0,
                Some(dep) => step_to_layer.get(&dep).copied().unwrap_or(0) + 1,
            };
            while layers.len() <= layer {
                layers.push(vec![]);
            }
            layers[layer].push(step.order);
            step_to_layer.insert(step.order, layer);
        }
        layers
    }
}

                    .await;

/// v1.2 E5: 扫活跃 map，移除 created_at + idle_timeout < now 的 agent
/// 返回移除数量
pub fn cleanup_idle(&self) -> usize {
    let Some(timeout) = self.idle_timeout else {
        return 0; // idle_timeout = None 不清理（v1.1 兼容）
    };
    let now = std::time::SystemTime::now();
    let mut map = match self.active.lock() {
        Ok(g) => g,
        Err(_) => return 0,
    };
    let before = map.len();
    map.retain(|_, sa| {
        now.duration_since(sa.created_at)
            .map(|d| d < timeout)
            .unwrap_or(true) // duration_since 失败（时钟回退）→ 保留
    });
    before - map.len()
}
```

//! v1.2 F3: TUI 交互层实现（设计 §14.3）
//!
//! 布局（ratatui 2 行 + 1 状态栏 + 1 输入框）：
//! ┌─────────────────────────────────────────┐
//! │ Header: eflow | profile | cache hit rate │  ← 1 行
//! ├─────────────────────────────────────────┤
//! │                                         │
//! │ Messages / Events (滚动)                │  ← main 区
//! │                                         │
//! ├─────────────────────────────────────────┤
//! │ Status: "Ready" / "Working..."           │  ← 1 行
//! │ > [prompt]                              │  ← 1 行输入框
//! └─────────────────────────────────────────┘

use std::io::Stdout;
use std::sync::Arc;
use std::time::Duration;

use crossterm::event::{Event as CtEvent, EventStream, KeyCode, KeyEvent, KeyEventKind};
use futures_util::StreamExt;

use ratatui::layout::{Constraint, Direction, Layout, Rect};

use ratatui::text::{Line, Span};
use ratatui::widgets::{Block, Borders, List, ListItem, Paragraph};
use ratatui::Terminal;

use super::layer::InteractionLayer;

/// TUI 状态
///
/// v1.2 F3: `profile` / `cache_hit_rate` 需要明确初值（不能随便 ""），所以不 derive Default，
/// 改用 `initial()` 显式构造。`std::sync::Mutex`（不是 tokio）因为 event loop 是 sync。
struct TuiState {
    messages: Vec<String>,
    status: String,
    prompt_buffer: String,
    profile: String,
    cache_hit_rate: String,
}

impl TuiState {
    /// 启动时的初值
    fn initial() -> Self {
        Self {
            messages: vec!["TUI 启动 — 输入任务并按 Enter".into()],
            status: "Ready".into(),
            prompt_buffer: String::new(),
            profile: String::new(), // run() 启动时同步 block_on 填充
            cache_hit_rate: "n/a".into(),
        }
    }
}

pub struct TuiBackend {
    /// tick interval（refresh status bar）
    tick_rate: Duration,
    /// v1.2 F3 补充：F6 main.rs 在 async 上下文填好后注入
    initial_profile: String,
    initial_cache_hit_rate: String,
}

impl TuiBackend {
    #[must_use]
    pub fn new() -> Self {
        Self {
            tick_rate: Duration::from_millis(250),
            initial_profile: String::new(),
            initial_cache_hit_rate: "n/a".into(),
        }
    }

    /// v1.2 F3 补充：main.rs 在 async 上下文拿到 initial profile + cache stats
    /// 后，用这个构造器注入。F6 main.rs 改造用。
    #[must_use]
    pub fn with_initial(profile: String, cache_hit_rate: String) -> Self {
        Self {
            tick_rate: Duration::from_millis(250),
            initial_profile: profile,
            initial_cache_hit_rate: cache_hit_rate,
        }
    }

    /// 渲染当前 state 到 frame
    fn render(state: &TuiState, f: &mut ratatui::Frame) {
        let chunks = Layout::default()
            .direction(Direction::Vertical)
            .constraints([
                Constraint::Length(1),  // header
                Constraint::Min(3),     // messages
                Constraint::Length(1),  // status
                Constraint::Length(1),  // prompt
            ])
            .split(f.area());

        // Header
        let header = Paragraph::new(Line::from(vec![
            Span::raw("eflow | "),
            Span::styled(
                format!("profile: {}", state.profile),
                Style::default().fg(Color::Cyan),
            ),
            Span::raw(" | "),
            Span::styled(
                format!("L2 cache: {}", state.cache_hit_rate),
                Style::default().fg(Color::Yellow),
            ),
        ]));
        f.render_widget(header, chunks[0]);

        // Messages
        let items: Vec<ListItem> = state
            .messages
            .iter()
            .map(|m| ListItem::new(m.as_str()))
            .collect();
        let messages = List::new(items)
            .block(Block::default().borders(Borders::NONE));
        f.render_widget(messages, chunks[1]);

        // Status
        let status = Paragraph::new(state.status.as_str())
            .style(Style::default().add_modifier(Modifier::ITALIC));
        f.render_widget(status, chunks[2]);

        // Prompt
        let prompt = Paragraph::new(format!("> {}", state.prompt_buffer));
        f.render_widget(prompt, chunks[3]);
    }
}

impl Default for TuiBackend {
    fn default() -> Self {
        Self::new()
    }
}

impl InteractionLayer for TuiBackend {
    fn run(&self, _concierge: Arc<Mutex<Concierge>>, _events: EventChannel) {
        // F3 占位：F4 + F5 加事件循环 + 异步输入
        // 当前实现：启动 ratatii + 立即退出（仅验证骨架能跑）
        let backend = CrosstermBackend::new(std::io::stdout());
        let mut terminal = Terminal::new(backend).expect("terminal init");
        terminal.clear().expect("clear");
        let state = TuiState::initial();
        terminal
            .draw(|f| Self::render(&state, f))
            .expect("draw");
        // 不接事件循环 → 立即返回（F4 + F5 加）
    }
}

- [ ] **Step 3: 实现 TUI event loop（订阅 EventChannel + 渲染刷新）**

> **设计要点**：`TuiBackend::run` 是 **同步** 函数（`fn run(&self, ...)` 非 `async fn`），
> 因为 ratatui + crossterm 的事件循环是同步的（用 `crossterm::event::poll` 阻塞）。
> `TuiState` 用 **`std::sync::Mutex`**（同步）以便在 sync 上下文持有。
> `Concierge` 是 `Arc<tokio::sync::Mutex<Concierge>>`，在拿到 Enter 提交的命令后用
> `tokio::spawn` 异步派发。task 事件走 `event_rx.recv()` 异步，配合
> `crossterm::event::poll` 同步；用 `tokio::runtime::Handle::current()` 拿 runtime handle
> 跨过 sync/async 边界。

替换 `src/interaction/tui.rs` 的 `InteractionLayer::run` 实现：

```rust
impl InteractionLayer for TuiBackend {
    fn run(&self, concierge: Arc<Mutex<Concierge>>, events: EventChannel) {
        // 启动 ratatui
        let backend = CrosstermBackend::new(std::io::stdout());
        let mut terminal = Terminal::new(backend).expect("terminal init");
        let _ = terminal.clear();

        // 启用 raw mode + 隐藏光标
        let _ = crossterm::terminal::enable_raw_mode();
        let _ = crossterm::execute!(
            std::io::stdout(),
            crossterm::event::EnableMouseCapture,
            crossterm::cursor::Hide
        );

        // 状态：std::sync::Mutex 同步，sync event loop 直接 .lock()
        // profile / cache_hit_rate 由 F3 with_initial 注入（main.rs 在 async 上下文填好）
        let state = std::sync::Arc::new(std::sync::Mutex::new(TuiState {
            messages: vec!["TUI 启动 — 输入任务并按 Enter".into()],
            status: "Ready".into(),
            prompt_buffer: String::new(),
            profile: self.initial_profile.clone(),
            cache_hit_rate: self.initial_cache_hit_rate.clone(),
        }));

        // task 事件 channel
        let mut event_rx = events.subscribe();

        // 拿当前 tokio runtime handle（main 是 #[tokio::main]，handle 在 tokio context 存在）
        // 用 spawn 派发 Concierge.handle_input（不需 block_on）
        let rt = tokio::runtime::Handle::current();

        // 主循环：同步 poll 键盘 + 异步 select 任务事件
        let result: std::io::Result<()> = (|| loop {
            // 渲染
            let state_lock = state.lock().unwrap();
            terminal.draw(|f| Self::render(&state_lock, f))?;
            drop(state_lock);

            // 阻塞等 crossterm 事件（带 timeout 让 task 事件有处理机会）
            if crossterm::event::poll(self.tick_rate)? {
                if let Ok(CtEvent::Key(key)) = crossterm::event::read() {
                    if key.kind == KeyEventKind::Press {
                        match key.code {
                            KeyCode::Char('c')
                                if key.modifiers.contains(crossterm::event::KeyModifiers::CONTROL) =>
                            {
                                return Ok(()); // Ctrl+C 退出
                            }
                            KeyCode::Esc => return Ok(()),
                            // F5 处理 prompt：把 KeyCode 透传给 handle_input_key
                            kc => {
                                let mut s = state.lock().unwrap();
                                if let Some(cmd) = handle_input_key(&mut s, kc) {
                                    drop(s);
                                    // Concierge 异步派发
                                    let concierge = concierge.clone();
                                    rt.spawn(async move {
                                        let _ = concierge.lock().await.handle_input(cmd).await;
                                    });
                                }
                            }
                        }
                    }
                }
            }

            // 非阻塞尝试接收 task 事件（不阻塞主循环；超时由上面的 poll 控制）
            loop {
                let event = match event_rx.try_recv() {
                    Ok(e) => e,
                    Err(broadcast::error::TryRecvError::Empty) => break,
                    Err(broadcast::error::TryRecvError::Lagged(_)) => continue,
                    Err(broadcast::error::TryRecvError::Closed) => return Ok(()),
                };
                let mut s = state.lock().unwrap();
                match event {
                    Event::TaskStarted { description, .. } => {
                        s.messages.push(format!("▶ {description}"));
                        s.status = "Working...".into();
                    }
                    Event::TaskCompleted { summary, .. } => {
                        let short: String = summary.chars().take(200).collect();
                        s.messages.push(format!("✓ {short}"));
                        s.status = "Ready".into();
                    }
                    Event::TaskFailed { error, .. } => {
                        s.messages.push(format!("✗ {error}"));
                        s.status = "Failed".into();
                    }
                    Event::RiskEscalated { from, to, .. } => {
                        s.messages.push(format!("⚠ Risk: {:?} → {:?}", from, to));
                        s.status = "Risk escalated".into();
                    }
                    Event::UserInputRequired { prompt } => {
                        s.messages.push(format!("? {prompt}"));
                    }
                    Event::SystemShutdown => return Ok(()),
                }
            }
        })();

        // 清理
        let _ = crossterm::terminal::disable_raw_mode();
        let _ = crossterm::execute!(
            std::io::stdout(),
            crossterm::event::DisableMouseCapture,
            crossterm::cursor::Show
        );
        let _ = terminal.show_cursor();

        if let Err(e) = result {
            tracing::error!("TUI loop error: {e}");
        }
    }
}
```

/// 返回 Some(cmd) 表示用户提交了命令（Enter），调用方应把它喂给 Concierge
fn handle_input_key(state: &mut TuiState, code: KeyCode) -> Option<String> {
    match code {
        KeyCode::Char(c) => {
            state.prompt_buffer.push(c);
            None
        }
        KeyCode::Backspace => {
            state.prompt_buffer.pop();
            None
        }
        KeyCode::Enter => {
            let cmd = std::mem::take(&mut state.prompt_buffer);
            if !cmd.is_empty() {
                state.messages.push(format!("> {cmd}"));
                state.status = "Dispatching...".into();
                Some(cmd)
            } else {
                None
            }
        }
        _ => None,
    }
}

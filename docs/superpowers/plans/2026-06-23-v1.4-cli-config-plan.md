```rust
// src/cli/prompt.rs — 终端交互组件（纯文本，无 ratatui）

use crossterm::event::{self, Event, KeyCode, KeyEventKind};
use std::io::{stdout, Write};

#[derive(Debug, Clone)]
pub struct MenuItem {
    pub key: &'static str,
    pub label: String,
}

/// ↑↓ 选择菜单。返回选中项索引，None = 取消。
/// 支持 ↑↓ / j/k / 数字键 / Enter / Esc
pub fn select_menu(items: &[MenuItem]) -> Option<usize> {
    use crossterm::terminal;

    terminal::enable_raw_mode().ok()?;
    let mut selected = 0usize;

    loop {
        // 清屏/上移并重绘
        print!("\r"); // 回到行首
        for (i, item) in items.iter().enumerate() {
            let prefix = if i == selected { " >" } else { "  " };
            println!("{} {}. {}", prefix, item.key, item.label);
        }
        print!("\x1b[{}A", items.len()); // 上移 N 行
        stdout().flush().ok()?;

        match event::read() {
            Ok(Event::Key(ke)) if ke.kind == KeyEventKind::Press => match ke.code {
                KeyCode::Up | KeyCode::Char('k') => {
                    selected = selected.saturating_sub(1);
                }
                KeyCode::Down | KeyCode::Char('j') => {
                    if selected + 1 < items.len() {
                        selected += 1;
                    }
                }
                KeyCode::Enter => {
                    terminal::disable_raw_mode().ok()?;
                    // 清除菜单行
                    for _ in 0..items.len() {
                        print!("\x1b[K\x1b[1A"); // 清行 + 上移
                    }
                    print!("\x1b[J"); // 清剩余
                    stdout().flush().ok()?;
                    return Some(selected);
                }
                KeyCode::Esc => {
                    terminal::disable_raw_mode().ok()?;
                    for _ in 0..items.len() {
                        print!("\x1b[K\x1b[1A");
                    }
                    print!("\x1b[J");
                    stdout().flush().ok()?;
                    return None;
                }
                KeyCode::Char(c) if c.is_ascii_digit() => {
                    let n = c.to_digit(10).unwrap_or(0) as usize;
                    if n >= 1 && n <= items.len() {
                        selected = n - 1;
                        terminal::disable_raw_mode().ok()?;
                        for _ in 0..items.len() {
                            print!("\x1b[K\x1b[1A");
                        }
                        print!("\x1b[J");
                        stdout().flush().ok()?;
                        return Some(selected);
                    }
                }
                _ => {}
            },
            _ => {}
        }
    }
}
```

```rust
/// 普通文本输入
pub fn prompt_input(prompt: &str) -> String {
    use std::io::{stdin, BufRead};
    print!("{} ", prompt);
    stdout().flush().ok();
    let mut line = String::new();
    stdin().lock().read_line(&mut line).ok();
    line.trim().to_string()
}

/// 掩码密码输入（关回显）
pub fn prompt_password(prompt: &str) -> String {
    use crossterm::terminal;
    print!("{} ", prompt);
    stdout().flush().ok();
    terminal::disable_echo().ok();
    let mut line = String::new();
    std::io::stdin().read_line(&mut line).ok();
    terminal::enable_echo().ok();
    println!(); // 换行
    line.trim().to_string()
}
```

```rust
// src/cli/config.rs — CLI 文本交互式配置（6 步）

use crate::interaction::wizard::builtin::provider::PRESETS;
use crate::cli::prompt::{select_menu, MenuItem, prompt_input, prompt_password};

/// 运行 CLI 文本配置流程。返回退出码。
pub fn run() -> i32 {
    // Step 1: 欢迎
    println!("\n  ╔══ LLM 配置向导 ══╗\n");
    println!("  未检测到 LLM 配置，开始设置。\n");
    println!("  按 Enter 继续...");
    let mut buf = String::new();
    std::io::stdin().read_line(&mut buf).ok();

    // Step 2: 选厂商
    let items: Vec<MenuItem> = PRESETS.iter().map(|p| MenuItem {
        key: "",  // 序号由 select_menu 自动生成
        label: format!("{} ({:?})", p.display_name, p.protocol),
    }).collect();
    // 加自定义选项
    let all_items = [items, vec![MenuItem {
        key: "",
        label: "自定义（兼容 OpenAI / Anthropic）".into(),
    }]].concat();

    let sel = select_menu(&all_items);
    let sel = match sel {
        Some(s) => s,
        None => return 1,
    };

    let is_custom = sel == PRESETS.len();
    let preset = if !is_custom { Some(&PRESETS[sel]) } else { None };

    // Step 3: 填 Key
    let api_key = prompt_password("API Key:");

    // Step 4: 拉模型列表
    let mut models: Vec<String> = Vec::new();
    if let Some(p) = preset {
        models = p.preset_models.iter().map(|s| s.to_string()).collect();
        println!("  正在拉取可用模型...");
        let url = format!("{}/models", p.base_url.trim_end_matches('/'));
        let client = reqwest::blocking::Client::builder()
            .timeout(std::time::Duration::from_secs(10))
            .build();
        if let Ok(client) = client {
            let resp = client.get(&url)
                .header("Authorization", format!("Bearer {}", api_key))
                .send();
            if let Ok(resp) = resp {
                if let Ok(json) = resp.json::<serde_json::Value>() {
                    if let Some(data) = json.get("data").and_then(|d| d.as_array()) {
                        for m in data {
                            if let Some(id) = m.get("id").and_then(|id| id.as_str()) {
                                let id = id.to_string();
                                if !models.contains(&id) {
                                    models.push(id);
                                }
                            }
                        }
                    }
                }
            }
        }
    }

    // Step 5: 选模型
    let model_items: Vec<MenuItem> = models.iter().map(|m| MenuItem {
        key: "",
        label: m.clone(),
    }).collect();
    let model_sel = if !model_items.is_empty() {
        select_menu(&model_items).map(|i| models[i].clone())
    } else {
        Some(prompt_input("手填模型 ID:"))
    };
    let model = match model_sel {
        Some(m) => m,
        None => return 1,
    };

    // Step 6: 确认并写入
    if is_custom {
        // 自定义路径：手填 protocol + base_url + 模型
        let protocol = prompt_input("协议 (openai/anthropic):");
        let base_url = prompt_input("Base URL:");
        write_provider("custom", "Custom", &protocol, &base_url, &api_key, &model);
    } else if let Some(p) = preset {
        let protocol_str = match p.protocol {
            crate::infrastructure::llm::types::ProtocolKind::OpenaiCompatible => "openai",
            crate::infrastructure::llm::types::ProtocolKind::AnthropicCompatible => "anthropic",
        };
        write_provider(p.id, p.display_name, protocol_str, p.base_url, &api_key, &model);
    }

    println!("\n  ✓ 配置已保存到 ~/.eflow/providers/");
    println!("  重新运行 eflow 开始使用。\n");
    0
}

fn write_provider(id: &str, name: &str, protocol: &str, base_url: &str, api_key: &str, model: &str) {
    use std::io::Write;

    let home = dirs::config_dir()
        .unwrap_or_else(|| std::path::PathBuf::from("."))
        .join("eflow").join("providers");
    std::fs::create_dir_all(&home).ok();

    let path = home.join(format!("{}.yaml", id));
    let yaml = format!(
        "id: {}\ndisplay_name: \"{}\"\nprotocol: {}_compatible\nbase_url: \"{}\"\napi_key: \"${{{}}}}\"\ndefault_model: \"{}\"\n",
        id, name, protocol, base_url, id.to_uppercase(), model
    );
    let mut file = std::fs::File::create(&path).ok();
    if let Some(ref mut f) = file {
        f.write_all(yaml.as_bytes()).ok();
    }

    // 更新 eflow.yaml routing
    update_eflow_yaml(id);
}

fn update_eflow_yaml(provider_id: &str) {
    let path = std::path::Path::new("eflow.yaml");
    if !path.exists() { return; }
    // 简化实现：直接替换 routing 块
    let content = std::fs::read_to_string(path).unwrap_or_default();
    let lines: Vec<&str> = content.lines().collect();
    let mut output = Vec::new();
    let mut in_routing = false;
    for line in &lines {
        if line.trim() == "routing:" {
            in_routing = true;
            output.push(*line);
            output.push(&format!("    strong: {}", provider_id));
            output.push(&format!("    medium: {}", provider_id));
            output.push(&format!("    light: {}", provider_id));
        } else if in_routing {
            if line.starts_with("  ") || line.starts_with("    ") {
                // skip old routing lines
            } else {
                in_routing = false;
                output.push(*line);
            }
        } else {
            output.push(*line);
        }
    }
    std::fs::write(path, output.join("\n")).ok();
}
```

```rust
        Some(Command::Init) => {
            let exit_code = eflow::cli::init::run();
            std::process::exit(exit_code);

            action: SessionAction::Start { config, lang },
```

```rust
/// 检查是否已配置 LLM provider
fn check_llm_configured() -> bool {
    let provider_dir = dirs::config_dir()
        .map(|p| p.join("eflow").join("providers"))
        .unwrap_or_else(|| std::path::PathBuf::from("./providers"));
    if let Ok(entries) = std::fs::read_dir(&provider_dir) {
        for entry in entries.flatten() {
            if entry.path().extension().map(|e| e == "yaml").unwrap_or(false) {
                return true;
            }
        }
    }
    false
}
```

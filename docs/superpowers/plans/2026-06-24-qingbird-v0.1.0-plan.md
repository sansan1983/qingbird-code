```rust
//! Retry logic — 指数退避

use std::time::Duration;
use crate::common::error::Result;
use crate::infrastructure::llm::types::ChatRequest;
use super::tier::TierRouter;

/// 指数退避重试（最多 max_retries 次，间隔 backoff_ms * 2^n）
pub async fn chat_with_retry(
    router: &TierRouter,
    tier: crate::common::types::ModelTier,
    request: ChatRequest,
    max_retries: u8,
    backoff_ms: u64,
) -> Result<super::types::ChatResponse> {
    let mut attempts = 0u8;
    let mut backoff = backoff_ms;
    loop {
        match router.chat(tier, request.clone()).await {
            Ok(resp) => return Ok(resp),
            Err(e) if attempts < max_retries => {
                attempts += 1;
                tokio::time::sleep(Duration::from_millis(backoff)).await;
                backoff *= 2;
            }
            Err(e) => return Err(e),
        }
    }
}
```

//! L1 in-memory LRU cache — 近期缓存最近 visible 的 LLM 响应

use std::num::NonZeroUsize;
use std::sync::Mutex;
use lru::LruCache;
use super::cache::CacheValue;

pub struct L1Cache {
    cache: Mutex<LruCache<u64, CacheValue>>,
    hits: u64,
    misses: u64,
}

impl L1Cache {
    pub fn new(capacity: usize) -> Self {
        Self {
            cache: Mutex::new(LruCache::new(NonZeroUsize::new(capacity).unwrap())),
            hits: 0,
            misses: 0,
        }
    }

    pub fn get(&self, hash: u64) -> Option<CacheValue> {
        let mut cache = self.cache.lock().unwrap();
        cache.get(&hash).cloned()
    }

    pub fn put(&self, hash: u64, value: CacheValue) {
        let mut cache = self.cache.lock().unwrap();
        cache.put(hash, value);
    }
}

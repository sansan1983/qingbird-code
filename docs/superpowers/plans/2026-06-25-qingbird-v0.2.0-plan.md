use async_trait::async_trait;
use std::path::Path;
use std::sync::LazyLock;
use walkdir::WalkDir;

use super::registry::{Tool, ToolDefinition, ToolOutput};
use crate::common::error::{EflowError, Result};
use crate::common::types::RiskLevel;
use rust_i18n::t;

/// 列出目录内容
pub struct ListDirTool;

#[async_trait]
impl Tool for ListDirTool {
    fn definition(&self) -> ToolDefinition {
        ToolDefinition {
            name: "list_dir".into(),
            description: t!("tool_desc_list_dir").into(),
            parameters: serde_json::json!({
                "type": "object",
                "properties": {
                    "path": {"type": "string", "description": "目录路径"},
                    "recursive": {"type": "boolean", "description": "是否递归列出子目录"}
                },
                "required": ["path"]
            }),
            risk_level: RiskLevel::L0,
        }
    }

    async fn execute(&self, params: serde_json::Value) -> Result<ToolOutput> {
        let path = params["path"].as_str().ok_or_else(|| {
            EflowError::Tool(t!("err_tool_missing_param", name = "path").to_string())
        })?;
        let recursive = params["recursive"].as_bool().unwrap_or(false);

        let root = Path::new(path);
        if !root.exists() {
            return Err(EflowError::Tool(
                t!("err_tool_invalid_path", path = path).to_string(),
            ));
        }
        if !root.is_dir() {
            return Err(EflowError::Tool(
                t!("err_tool_not_a_dir", path = path).to_string(),
            ));
        }

        let mut entries: Vec<String> = Vec::new();

        if recursive {
            for entry in WalkDir::new(root)
                .follow_links(false)
                .into_iter()
                .filter_map(std::result::Result::ok)
            {
                let entry_path = entry.path();
                let display = entry_path.display();
                if entry.file_type().is_dir() {
                    entries.push(format!("[DIR]  {}/", display));
                } else if entry.file_type().is_file() {
                    let len = entry.metadata().map(|m| m.len()).unwrap_or(0);
                    entries.push(format!("[FILE] {}  {} bytes", display, len));
                }
            }
        } else {
            let mut read_dir = tokio::fs::read_dir(root).await.map_err(|e| {
                EflowError::Tool(
                    t!("err_tool_read_dir", path = path, msg = e.to_string()).to_string(),
                )
            })?;

            let mut child_entries: Vec<(String, u64, bool)> = Vec::new();
            while let Some(child) = read_dir.next_entry().await.map_err(|e| {
                EflowError::Tool(
                    t!("err_tool_read_dir", path = path, msg = e.to_string()).to_string(),
                )
            })? {
                let file_type = child.file_type().await.unwrap_or_default();
                let name = child.file_name().to_string_lossy().to_string();
                let len = child.metadata().map(|m| m.len()).unwrap_or(0);
                child_entries.push((name, len, file_type.is_dir()));
            }

            // 目录在前，文件在后，各自按名称排序
            child_entries.sort_by(|a, b| a.0.cmp(&b.0));
            for (name, len, is_dir) in child_entries {
                if is_dir {
                    entries.push(format!("[DIR]  {}/", name));
                } else {
                    entries.push(format!("[FILE] {}  {} bytes", name, len));
                }
            }
        }

        let count = entries.len();
        let body = t!("status_list_dir_header", path = path, count = count).to_string()
            + "\n\n"
            + &entries.join("\n");

        Ok(ToolOutput {
            success: true,
            content: body,
            metadata: Some(serde_json::json!({"entries": count})),
        })
    }
}

/// 按 glob 模式匹配文件
pub struct GlobTool;

#[async_trait]
impl Tool for GlobTool {
    fn definition(&self) -> ToolDefinition {
        ToolDefinition {
            name: "glob".into(),
            description: t!("tool_desc_glob").into(),
            parameters: serde_json::json!({
                "type": "object",
                "properties": {
                    "pattern": {"type": "string", "description": "glob 模式，如 **/*.rs"},
                    "root": {"type": "string", "description": "搜索根目录，默认 '.'"}
                },
                "required": ["pattern"]
            }),
            risk_level: RiskLevel::L0,
        }
    }

    async fn execute(&self, params: serde_json::Value) -> Result<ToolOutput> {
        let pattern = params["pattern"].as_str().ok_or_else(|| {
            EflowError::Tool(t!("err_tool_missing_param", name = "pattern").to_string())
        })?;
        let root = params["root"].as_str().unwrap_or(".");

        let root_path = Path::new(root);
        if !root_path.exists() {
            return Err(EflowError::Tool(
                t!("err_tool_invalid_path", path = root).to_string(),
            ));
        }

        let mut results: Vec<String> = Vec::new();
        let max_results = 200;

        for entry in WalkDir::new(root_path)
            .follow_links(false)
            .into_iter()
            .filter_map(std::result::Result::ok)
        {
            if !entry.file_type().is_file() {
                continue;
            }

            let relative = entry
                .path()
                .strip_prefix(root_path)
                .unwrap_or(entry.path())
                .to_string_lossy()
                .replace('\\', "/");

            if glob_match(pattern, &relative) {
                results.push(entry.path().display().to_string());
                if results.len() >= max_results {
                    break;
                }
            }
        }

        let count = results.len();
        let truncated = count >= max_results;
        let body = if results.is_empty() {
            t!("status_no_match", pattern = pattern).to_string()
        } else {
            let header = if truncated {
                t!("status_glob_truncated", count = count, pattern = pattern)
            } else {
                t!("status_glob_header", count = count, pattern = pattern)
            };
            header.to_string() + "\n\n" + &results.join("\n")
        };

        Ok(ToolOutput {
            success: true,
            content: body,
            metadata: Some(serde_json::json!({
                "matches": count,
                "truncated": truncated,
            })),
        })
    }
}

/// 简易 glob 匹配：将 glob 模式转为正则表达式
///
/// 支持:
/// - `*` — 匹配单层路径内的任意字符（不含 /）
/// - `**` — 匹配跨路径的任意字符
/// - `?` — 匹配单字符（不含 /）
fn glob_match(pattern: &str, path: &str) -> bool {
    let path = path.replace('\\', "/");

    // 手工构建正则，避免 regex_lite::escape + 替换的顺序问题
    let mut re_str = String::from("^");
    let mut chars = pattern.chars().peekable();

    while let Some(c) = chars.next() {
        match c {
            '*' if chars.peek() == Some(&'*') => {
                re_str.push_str(".*");
                chars.next(); // 消费第二个 *
            }
            '*' => re_str.push_str("[^/]*"),
            '?' => re_str.push_str("[^/]"),
            '.' => re_str.push_str("\\."),
            // 转义其他正则特殊字符
            '+' | '^' | '$' | '(' | ')' | '[' | ']' | '{' | '}' | '|' | '\\' => {
                re_str.push('\\');
                re_str.push(c);
            }
            c => re_str.push(c),
        }
    }
    re_str.push('$');

    if let Ok(re) = regex_lite::Regex::new(&re_str) {
        re.is_match(&path)
    } else {
        false
    }
}

use async_trait::async_trait;

use super::registry::{Tool, ToolDefinition, ToolOutput};
use crate::common::error::{EflowError, Result};
use crate::common::types::RiskLevel;
use rust_i18n::t;

/// HTTP GET 请求工具
pub struct HttpGetTool;

#[async_trait]
impl Tool for HttpGetTool {
    fn definition(&self) -> ToolDefinition {
        ToolDefinition {
            name: "http_get".into(),
            description: t!("tool_desc_http_get").into(),
            parameters: serde_json::json!({
                "type": "object",
                "properties": {
                    "url": {"type": "string", "description": "目标 URL"},
                    "timeout_secs": {"type": "integer", "description": "超时秒数，默认 30"}
                },
                "required": ["url"]
            }),
            risk_level: RiskLevel::L1,
        }
    }

    async fn execute(&self, params: serde_json::Value) -> Result<ToolOutput> {
        let url = params["url"].as_str().ok_or_else(|| {
            EflowError::Tool(t!("err_tool_missing_param", name = "url").to_string())
        })?;

        // 检查 URL 合法性
        if !url.starts_with("http://") && !url.starts_with("https://") {
            return Err(EflowError::Tool(
                t!("err_tool_invalid_url", url = url).to_string(),
            ));
        }

        let timeout_secs = params["timeout_secs"].as_u64().unwrap_or(30);
        let timeout = std::time::Duration::from_secs(timeout_secs);

        let client = reqwest::Client::builder()
            .timeout(timeout)
            .build()
            .map_err(|e| {
                EflowError::Tool(
                    t!("err_tool_http_client", msg = e.to_string()).to_string(),
                )
            })?;

        let response = client.get(url).send().await.map_err(|e| {
            EflowError::Tool(
                t!("err_tool_http_get", url = url, msg = e.to_string()).to_string(),
            )
        })?;

        let status = response.status().as_u16();
        let content_type = response
            .headers()
            .get(reqwest::header::CONTENT_TYPE)
            .and_then(|v| v.to_str().ok())
            .unwrap_or("")
            .to_string();

        // 只读取前 1MB 内容防止 OOM
        const MAX_BODY: usize = 1_048_576;
        let body = response.text().await.map_err(|e| {
            EflowError::Tool(
                t!("err_tool_http_body", msg = e.to_string()).to_string(),
            )
        })?;
        let truncated = body.len() > MAX_BODY;
        let display_body = if truncated {
            format!("{}...\n(response truncated at {} bytes)", &body[..MAX_BODY], MAX_BODY)
        } else {
            body
        };

        let content = format!(
            "HTTP {} ({})\n\n{}",
            status, content_type, display_body
        );

        Ok(ToolOutput {
            success: status < 500,
            content,
            metadata: Some(serde_json::json!({
                "status_code": status,
                "content_type": content_type,
                "truncated": truncated,
            })),
        })
    }
}

use async_trait::async_trait;
use std::path::Path;

use super::registry::{Tool, ToolDefinition, ToolOutput};
use crate::common::error::{EflowError, Result};
use crate::common::types::RiskLevel;
use rust_i18n::t;

/// 精准替换文件中的文本（需唯一匹配）
pub struct EditFileTool;

#[async_trait]
impl Tool for EditFileTool {
    fn definition(&self) -> ToolDefinition {
        ToolDefinition {
            name: "edit_file".into(),
            description: t!("tool_desc_edit_file").into(),
            parameters: serde_json::json!({
                "type": "object",
                "properties": {
                    "path": {"type": "string", "description": "文件路径"},
                    "old_string": {"type": "string", "description": "被替换的文本（需唯一匹配）"},
                    "new_string": {"type": "string", "description": "替换后的文本"}
                },
                "required": ["path", "old_string", "new_string"]
            }),
            risk_level: RiskLevel::L1,
        }
    }

    async fn execute(&self, params: serde_json::Value) -> Result<ToolOutput> {
        let path = params["path"].as_str().ok_or_else(|| {
            EflowError::Tool(t!("err_tool_missing_param", name = "path").to_string())
        })?;
        let old_string = params["old_string"].as_str().ok_or_else(|| {
            EflowError::Tool(t!("err_tool_missing_param", name = "old_string").to_string())
        })?;
        let new_string = params["new_string"].as_str().ok_or_else(|| {
            EflowError::Tool(t!("err_tool_missing_param", name = "new_string").to_string())
        })?;

        let file_path = Path::new(path);
        if !file_path.exists() {
            return Err(EflowError::Tool(
                t!("err_tool_invalid_path", path = path).to_string(),
            ));
        }

        let content = tokio::fs::read_to_string(file_path).await.map_err(|e| {
            EflowError::Tool(
                t!("err_tool_read_file", path = path, msg = e.to_string()).to_string(),
            )
        })?;

        // 检查匹配次数
        let count = content.matches(old_string).count();
        if count == 0 {
            return Err(EflowError::Tool(
                t!("err_tool_edit_not_found", old = old_string).to_string(),
            ));
        }
        if count > 1 {
            // 显示前 3 处上下文帮助调试
            let mut contexts: Vec<String> = Vec::new();
            for (i, line) in content.lines().enumerate() {
                if line.contains(old_string) {
                    contexts.push(format!("  line {}: {}", i + 1, line));
                    if contexts.len() >= 3 {
                        break;
                    }
                }
            }
            return Err(EflowError::Tool(
                t!(
                    "err_tool_edit_ambiguous",
                    count = count,
                    old = old_string,
                    contexts = contexts.join("\n")
                )
                .to_string(),
            ));
        }

        // 唯一匹配，安全替换
        let new_content = content.replace(old_string, new_string);
        tokio::fs::write(file_path, new_content).await.map_err(|e| {
            EflowError::Tool(
                t!("err_tool_write_file", path = path, msg = e.to_string()).to_string(),
            )
        })?;

        let bytes = content.len();
        Ok(ToolOutput {
            success: true,
            content: t!("status_edit_success", path = path).to_string(),
            metadata: Some(serde_json::json!({
                "file_size": bytes,
            })),
        })
    }
}

`src/cli/output.rs` (文件末尾, 加在已有的 CliOutput 之后):
```rust
/// CLI 进度指示器 — 在 stderr 上显示旋转动画
///
/// ADR-V0.2-002: 输出到 stderr，保持 stdout 干净
pub struct Spinner {
    stop_tx: Option<tokio::sync::oneshot::Sender<()>>,
}

impl Spinner {
    /// 启动 spinner（自动在 stderr 上开始动画，返回句柄）
    pub fn start(message: &str) -> Self {
        let frames = ["⠋", "⠙", "⠹", "⠸", "⠼", "⠴", "⠦", "⠧", "⠇", "⠏"];
        let (tx, mut rx) = tokio::sync::oneshot::channel::<()>();
        let msg = message.to_string();

        tokio::spawn(async move {
            let mut i: usize = 0;
            loop {
                tokio::select! {
                    _ = &mut rx => {
                        // 清除 spinner 行
                        use std::io::Write;
                        let _ = write!(std::io::stderr(), "\r\x1b[K");
                        let _ = std::io::stderr().flush();
                        break;
                    }
                    _ = tokio::time::sleep(std::time::Duration::from_millis(120)) => {
                        use std::io::Write;
                        let _ = write!(
                            std::io::stderr(),
                            "\r{} {}",
                            frames[i % frames.len()],
                            msg
                        );
                        let _ = std::io::stderr().flush();
                        i += 1;
                    }
                }
            }
        });

        Self { stop_tx: Some(tx) }
    }

    /// 停止 spinner，清除输出行
    pub fn stop(mut self) {
        if let Some(tx) = self.stop_tx.take() {
            let _ = tx.send(());
        }
    }
}

impl Drop for Spinner {
    fn drop(&mut self) {
        if let Some(tx) = self.stop_tx.take() {
            let _ = tx.send(());
        }
    }
}
```

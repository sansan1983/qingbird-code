# eflow v1.0 内核实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建 eflow v1.0 内核——一个可运行的端到端 Agent 框架骨架：用户输入任务 → Concierge 识别意图 → Orchestrator 分解 → 三角色管线段 (Decisioner→Executor→Feedbacker) 执行 → 返回结果。

**Architecture:** 自底向上构建，从零依赖的类型定义开始，逐步叠加基础设施(Event/LLM/Memory/Context)，再组装能力层(Blackboard/三角色/Tools)，最后串通编排层(Orchestrator/Concierge)和交互层(CLI)。每层编译通过后再进入下一层。

**Tech Stack:** Rust 2024 edition, tokio async runtime, clap CLI, serde YAML, rusqlite, reqwest HTTP, async-trait, uuid, sha2, tracing

**Spec reference:** `eflow-Framework-Design-v4.0.md`

```rust
/// 加载配置文件
pub fn load_config(path: &Path) -> crate::common::error::Result<EflowConfig> {
    let content = std::fs::read_to_string(path)
        .map_err(|e| crate::common::error::EflowError::Config(
            format!("Failed to read config file {}: {}", path.display(), e)
        ))?;

    // 展开环境变量 ${VAR_NAME}
    let expanded = expand_env_vars(&content);

    let config: EflowConfig = serde_yaml::from_str(&expanded)
        .map_err(|e| crate::common::error::EflowError::Config(
            format!("Failed to parse config: {}", e)
        ))?;

    Ok(config)
}

/// 寻找配置文件：当前目录 → ~/.eflow/ → /etc/eflow/
pub fn find_config() -> Option<std::path::PathBuf> {
    let candidates = [
        std::path::PathBuf::from("eflow.yaml"),
        dirs_config_path().join("eflow.yaml"),
        std::path::PathBuf::from("/etc/eflow/eflow.yaml"),
    ];

    for candidate in &candidates {
        if candidate.exists() {
            return Some(candidate.clone());
        }
    }
    None
}

fn dirs_config_path() -> std::path::PathBuf {
    if let Some(home) = std::env::var_os("HOME") {
        std::path::PathBuf::from(home).join(".eflow")
    } else if let Some(userprofile) = std::env::var_os("USERPROFILE") {
        std::path::PathBuf::from(userprofile).join(".eflow")
    } else {
        std::path::PathBuf::from(".")
    }
}

fn expand_env_vars(input: &str) -> String {
    let re = regex_lite::Regex::new(r"\$\{(\w+)\}").unwrap();
    re.replace_all(input, |caps: &regex_lite::Captures| {
        let var_name = &caps[1];
        std::env::var(var_name).unwrap_or_else(|_| format!("${{{}}}", var_name))
    }).to_string()
}
```

```rust
use tokio::sync::broadcast;
use uuid::Uuid;
use crate::common::types::RiskLevel;

/// 系统事件
#[derive(Debug, Clone)]
pub enum Event {
    TaskStarted { task_id: Uuid, description: String },
    TaskCompleted { task_id: Uuid, summary: String },
    TaskFailed { task_id: Uuid, error: String },
    RiskEscalated { task_id: Uuid, from: RiskLevel, to: RiskLevel },
    UserInputRequired { prompt: String },
    SystemShutdown,
}

/// 事件通道 — 基于 tokio::broadcast
pub struct EventChannel {
    tx: broadcast::Sender<Event>,
}

impl EventChannel {
    /// 创建新通道，缓冲区 256
    pub fn new() -> Self {
        let (tx, _) = broadcast::channel(256);
        Self { tx }
    }

    /// 发布事件（不等待，忽略无订阅者错误）
    pub fn publish(&self, event: Event) {
        let _ = self.tx.send(event);
    }

    /// 订阅事件流
    pub fn subscribe(&self) -> broadcast::Receiver<Event> {
        self.tx.subscribe()
    }
}

impl Default for EventChannel {
    fn default() -> Self {
        Self::new()
    }
}

impl Clone for EventChannel {
    fn clone(&self) -> Self {
        Self {
            tx: self.tx.clone(),
        }
    }
}
```

```rust
#[tokio::test]
async fn test_publish_and_receive() {

    let mut rx = channel.subscribe();

    let task_id = Uuid::new_v4();
    channel.publish(Event::TaskStarted {
        task_id,
        description: "test task".into(),
    });

    let received = rx.recv().await.unwrap();
    match received {
        Event::TaskStarted { task_id: id, description } => {

            assert_eq!(description, "test task");
```

```rust
        }
        if status == reqwest::StatusCode::TOO_MANY_REQUESTS {

        }

        let json: Value = response.json().await
            .map_err(|e| EflowError::LlmProvider(format!("JSON parse error: {}", e)))?;

        let content = json["content"]
            .as_array()
            .and_then(|blocks| {
                blocks.iter()
                    .filter(|b| b["type"] == "text")
                    .map(|b| b["text"].as_str().unwrap_or(""))
                    .collect::<Vec<_>>()
                    .join("\n")
            })
            .unwrap_or_default();

        let input_tokens = json["usage"]["input_tokens"].as_u64().unwrap_or(0) as u32;
        let output_tokens = json["usage"]["output_tokens"].as_u64().unwrap_or(0) as u32;

        Ok(ChatResponse {
            content,
            tool_calls: None, // Anthropic tool calls parsed separately
            usage: TokenUsage { input_tokens, output_tokens },
            finish_reason: json["stop_reason"].as_str().unwrap_or("unknown").into(),
        })
    }

    async fn chat_stream(
        &self,
        request: ChatRequest,
    ) -> Result<mpsc::Receiver<Result<ChatChunk>>> {
        let api_key = self.api_key.clone();
        let default_model = self.default_model.clone();

        let model = if request.model.is_empty() {
            default_model
        } else {
            request.model
        };

        let body = self.build_body(&ChatRequest {
            model: model.clone(),
            ..request
        });

        // 流式请求 body 加 stream: true
        let mut body = body;
        body["stream"] = serde_json::json!(true);

        let (tx, rx) = mpsc::channel(64);

        tokio::spawn(async move {
            let client = Client::new();
            let response = match client
                .post("https://api.anthropic.com/v1/messages")
                .header("x-api-key", &api_key)
                .header("anthropic-version", "2023-06-01")
                .header("content-type", "application/json")
                .json(&body)
                .send()
                .await
            {
                Ok(r) => r,
                Err(e) => {
                    let _ = tx.send(Err(EflowError::LlmProvider(format!("{}", e)))).await;
                    return;
                }
            };

            // 简化流式处理 — 逐行读取 SSE
            use futures_util::StreamExt;
            let mut stream = response.bytes_stream();
            while let Some(chunk_result) = stream.next().await {
                match chunk_result {
                    Ok(bytes) => {
                        let text = String::from_utf8_lossy(&bytes);
                        for line in text.lines() {
                            if line.starts_with("data: ") {
                                let data = &line[6..];
                                if data == "[DONE]" { continue; }
                                if let Ok(json) = serde_json::from_str::<Value>(data) {
                                    let content_delta = json["delta"]["text"]
                                        .as_str().map(|s| s.to_string());
                                    let _ = tx.send(Ok(ChatChunk {
                                        content_delta,
                                        tool_call_delta: None,
                                    })).await;
                                }
                            }
                        }
                    }
                    Err(_) => break,
                }
            }
        });

        Ok(rx)
    }

    fn supports_prefix_cache(&self) -> bool {
        true // Anthropic 支持 prompt caching
    }

    fn name(&self) -> &str {
        "anthropic"
    }
}
```

```rust
use std::collections::HashMap;
use std::sync::Arc;

use crate::common::error::{EflowError, Result};
use crate::common::types::ModelTier;
use crate::infrastructure::config::EflowConfig;

use super::types::*;
use super::anthropic::AnthropicProvider;
use super::openai::OpenAiProvider;

/// LLM Router — 统一入口，按 ModelTier 路由到具体 Provider
pub struct LlmRouter {
    providers: HashMap<String, Arc<dyn LlmProvider>>,
    routing: HashMap<ModelTier, String>, // ModelTier → provider_name
    rate_limit_counters: HashMap<String, u32>,
}

impl LlmRouter {
    /// 从配置创建 Router
    pub fn from_config(config: &EflowConfig) -> Result<Self> {
        let mut providers: HashMap<String, Arc<dyn LlmProvider>> = HashMap::new();

        // 加载 Anthropic
        if let Some(ref anthro) = config.llm.providers.anthropic {
            if !anthro.api_key.is_empty() && !anthro.api_key.starts_with("${") {
                let provider = AnthropicProvider::new(
                    anthro.api_key.clone(),
                    anthro.default_model.clone(),
                );
                providers.insert("anthropic".into(), Arc::new(provider));
            }
        }

        // 加载 OpenAI
        if let Some(ref openai) = config.llm.providers.openai {
            if !openai.api_key.is_empty() && !openai.api_key.starts_with("${") {
                let provider = OpenAiProvider::new(
                    openai.api_key.clone(),
                    openai.default_model.clone(),
                );
                providers.insert("openai".into(), Arc::new(provider));
            }
        }

        if providers.is_empty() {
            return Err(EflowError::Config(
                "No LLM providers configured. Set ANTHROPIC_API_KEY or OPENAI_API_KEY.".into()
            ));
        }

        // 构建路由表
        let mut routing = HashMap::new();
        let strong = config.llm.routing.strong.clone();
        let medium = config.llm.routing.medium.clone();
        let light = config.llm.routing.light.clone();

        // 只在 provider 存在时添加路由
        if providers.contains_key(&strong) {
            routing.insert(ModelTier::Strong, strong);
        }
        if providers.contains_key(&medium) {
            routing.insert(ModelTier::Medium, medium);
        }
        if providers.contains_key(&light) {
            routing.insert(ModelTier::Light, light);
        }

        // 回退：如果指定 tier 的 provider 不存在，使用任意可用的
        for tier in [ModelTier::Strong, ModelTier::Medium, ModelTier::Light] {
            if !routing.contains_key(&tier) {
                if let Some(name) = providers.keys().next() {
                    routing.insert(tier, name.clone());
                }
            }
        }

        Ok(Self {
            providers,
            routing,
            rate_limit_counters: HashMap::new(),
        })
    }

    /// 按 ModelTier 路由调用
    pub async fn chat(
        &mut self,
        tier: ModelTier,
        request: ChatRequest,
    ) -> Result<ChatResponse> {
        let provider_name = self.routing.get(&tier)
            .ok_or_else(|| EflowError::Internal(format!("No provider for tier {:?}", tier)))?
            .clone();

        let provider = self.providers.get(&provider_name)
            .ok_or_else(|| EflowError::Internal(format!("Provider '{}' not found", provider_name)))?
            .clone();

        match provider.chat(request).await {
            Ok(response) => {
                // 成功后清除限流计数
                self.rate_limit_counters.remove(&provider_name);
                Ok(response)
            }
            Err(EflowError::RateLimited(_)) => {
                let count = self.rate_limit_counters
                    .entry(provider_name.clone())
                    .or_insert(0);
                *count += 1;

                if *count >= 5 {
                    // 轻度降级：尝试降级到下一级模型
                    self.try_degraded_call(tier, provider_name, *count).await
                } else {
                    Err(EflowError::RateLimited(provider_name))
                }
            }
            Err(e) => Err(e),
        }
    }

    /// 当主 provider 被限流时尝试降级
    async fn try_degraded_call(
        &mut self,
        _original_tier: ModelTier,
        failed_provider: String,
        rate_limit_count: u32,
    ) -> Result<ChatResponse> {
        // 找另一个可用 provider
        let fallback = self.providers.keys()
            .find(|name| **name != failed_provider)
            .cloned();

        if let Some(fallback_name) = fallback {
            if rate_limit_count >= 10 {
                // 中度降级：所有 provider 都被限流
                return Err(EflowError::RateLimited(
                    format!("All providers rate limited (attempts: {})", rate_limit_count)
                ));
            }
            let provider = self.providers.get(&fallback_name).unwrap().clone();
            // 使用默认模型的简单请求重试
            let request = ChatRequest::new(
                provider.name(),
                vec![Message::user("retry")],
            );
            provider.chat(request).await
        } else {
            Err(EflowError::RateLimited(
                "No fallback provider available".into()
            ))
        }
    }

    /// 获取 provider 名称
    pub fn provider_for(&self, tier: ModelTier) -> Option<&str> {
        self.routing.get(&tier).map(|s| s.as_str())
    }
}
```

```rust
use std::time::SystemTime;
use indexmap::IndexMap;
use uuid::Uuid;

use super::manager::{MemoryEntry, MemoryManager, RecallScope};
use crate::common::error::Result;
use crate::common::types::Importance;

/// 工作记忆 — 内存 IndexMap，会话结束销毁
pub struct WorkingMemory {
    entries: IndexMap<Uuid, MemoryEntry>,
    max_entries: usize,
}

impl WorkingMemory {
    pub fn new(max_entries: usize) -> Self {
        Self {
            entries: IndexMap::new(),
            max_entries,
        }
    }
}

impl MemoryManager for WorkingMemory {
    fn remember(&mut self, mut entry: MemoryEntry) -> Result<Uuid> {
        if entry.id.is_nil() {
            entry.id = Uuid::new_v4();
        }
        entry.created_at = SystemTime::now();
        entry.last_accessed_at = SystemTime::now();

        let id = entry.id;
        self.entries.insert(id, entry);

        // LRU 淘汰
        if self.entries.len() > self.max_entries {
            self.entries.shift_remove_index(0);
        }

        Ok(id)
    }

    fn recall(&self, query: &str, _scope: RecallScope, limit: u8) -> Result<Vec<MemoryEntry>> {
        let query_lower = query.to_lowercase();
        let results: Vec<MemoryEntry> = self.entries.values()
            .rev()
            .filter(|e| {
                e.content.to_lowercase().contains(&query_lower)
                    || e.tags.iter().any(|t| t.to_lowercase().contains(&query_lower))
            })
            .take(limit as usize)
            .cloned()
            .collect();
        Ok(results)
    }

    fn recall_since(&self, since: SystemTime, _scope: RecallScope) -> Result<Vec<MemoryEntry>> {
        let results: Vec<MemoryEntry> = self.entries.values()
            .rev()
            .filter(|e| e.created_at >= since)
            .cloned()
            .collect();
        Ok(results)
    }

    fn forget(&mut self, id: Uuid) -> Result<()> {
        self.entries.shift_remove(&id);
        Ok(())
    }

    fn cleanup(&mut self) -> Result<u32> {
        let before = self.entries.len();
        let now = SystemTime::now();
        self.entries.retain(|_, e| {
            if let Some(ttl) = e.ttl {
                if e.importance == Importance::Low {
                    return now.duration_since(e.created_at).unwrap_or_default() < ttl;
                }
            }
            true
        });
        Ok((before - self.entries.len()) as u32)
    }

    fn session_summary(&self) -> Result<String> {
        let entries: Vec<String> = self.entries.values()
            .rev()
            .take(20)
            .map(|e| format!("- {}", &e.content[..e.content.len().min(200)]))
            .collect();
        Ok(entries.join("\n"))
    }
}
```

```rust
                ON memories(created_at);
            CREATE INDEX IF NOT EXISTS idx_memories_category
                ON memories(category);
            CREATE VIRTUAL TABLE IF NOT EXISTS memories_fts USING fts5(
                content, tags, content='memories', content_rowid='rowid'
            );
            CREATE TRIGGER IF NOT EXISTS memories_ai AFTER INSERT ON memories BEGIN
                INSERT INTO memories_fts(rowid, content, tags)
                VALUES (new.rowid, new.content, new.tags);

impl MemoryManager for ProjectMemory {
    fn remember(&mut self, mut entry: MemoryEntry) -> Result<Uuid> {
        if entry.id.is_nil() {
            entry.id = Uuid::new_v4();
        }
        let now = SystemTime::now();
        entry.created_at = now;
        entry.last_accessed_at = now;

        let tags_json = serde_json::to_string(&entry.tags).unwrap_or_default();
        let created_ts = now.duration_since(SystemTime::UNIX_EPOCH)
            .unwrap_or_default().as_secs() as i64;

        self.conn.execute(
            "INSERT OR REPLACE INTO memories (id, content, raw_content, category, importance,
             created_at, last_accessed_at, ttl_secs, tags)
             VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7, ?8, ?9)",
            params![
                entry.id.to_string(),
                entry.content,
                entry.raw_content,
                format!("{:?}", entry.category),
                format!("{:?}", entry.importance),
                created_ts,
                created_ts,
                entry.ttl.map(|d| d.as_secs() as i64),
                tags_json,
            ],
        ).map_err(|e| EflowError::Memory(format!("insert: {}", e)))?;

        Ok(entry.id)
    }

    fn recall(&self, query: &str, _scope: RecallScope, limit: u8) -> Result<Vec<MemoryEntry>> {
        let mut stmt = self.conn.prepare(
            "SELECT m.id, m.content, m.raw_content, m.category, m.importance,
                    m.created_at, m.last_accessed_at, m.ttl_secs, m.tags
             FROM memories m
             INNER JOIN memories_fts fts ON m.rowid = fts.rowid
             WHERE memories_fts MATCH ?1
             ORDER BY m.last_accessed_at DESC
             LIMIT ?2"
        ).map_err(|e| EflowError::Memory(format!("prepare: {}", e)))?;

        let rows = stmt.query_map(params![query, limit], |row| {
            Ok(row_to_entry(row).unwrap())
        }).map_err(|e| EflowError::Memory(format!("query: {}", e)))?;

        let mut entries = vec![];
        for row in rows {
            if let Ok(entry) = row {
                entries.push(entry);
            }
        }
        Ok(entries)
    }

    fn recall_since(&self, since: SystemTime, _scope: RecallScope) -> Result<Vec<MemoryEntry>> {
        let since_ts = since.duration_since(SystemTime::UNIX_EPOCH)
            .unwrap_or_default().as_secs() as i64;

        let mut stmt = self.conn.prepare(
            "SELECT id, content, raw_content, category, importance,
                    created_at, last_accessed_at, ttl_secs, tags
             FROM memories WHERE created_at >= ?1
             ORDER BY created_at DESC"
        ).map_err(|e| EflowError::Memory(format!("prepare: {}", e)))?;

        let rows = stmt.query_map(params![since_ts], |row| {
            Ok(row_to_entry(row).unwrap())
        }).map_err(|e| EflowError::Memory(format!("query: {}", e)))?;

        let mut entries = vec![];
        for row in rows {
            if let Ok(entry) = row {
                entries.push(entry);
            }
        }
        Ok(entries)
    }

    fn forget(&mut self, id: Uuid) -> Result<()> {
        self.conn.execute("DELETE FROM memories WHERE id = ?1", params![id.to_string()])
            .map_err(|e| EflowError::Memory(format!("delete: {}", e)))?;
        Ok(())
    }

    fn cleanup(&mut self) -> Result<u32> {
        let now_ts = SystemTime::now().duration_since(SystemTime::UNIX_EPOCH)
            .unwrap_or_default().as_secs() as i64;

        let count = self.conn.execute(
            "DELETE FROM memories WHERE
                importance = 'Low'
                AND ttl_secs IS NOT NULL
                AND (created_at + ttl_secs) < ?1",
            params![now_ts],
        ).map_err(|e| EflowError::Memory(format!("cleanup: {}", e)))?;

        Ok(count as u32)
    }

    fn session_summary(&self) -> Result<String> {
        // 返回最近 20 条记忆摘要
        let mut stmt = self.conn.prepare(
            "SELECT content FROM memories ORDER BY last_accessed_at DESC LIMIT 20"
        ).map_err(|e| EflowError::Memory(format!("prepare: {}", e)))?;

        let rows = stmt.query_map([], |row| {
            row.get::<_, String>(0)
        }).map_err(|e| EflowError::Memory(format!("query: {}", e)))?;

        let entries: Vec<String> = rows
            .filter_map(|r| r.ok())
            .map(|c| format!("- {}", &c[..c.len().min(200)]))
            .collect();

        Ok(entries.join("\n"))
    }
}
```

```rust
use uuid::Uuid;

use super::working::WorkingMemory;
use super::project::ProjectMemory;
use super::user::UserMemory;
use crate::common::error::Result;

/// 组合记忆管理器 — 串联三层记忆
pub struct CompositeMemory {
    pub working: WorkingMemory,
    pub project: ProjectMemory,
    pub user: UserMemory,
}

impl CompositeMemory {
    pub fn new(
        working_limit: usize,
        project_db: &std::path::Path,
        user_db: &std::path::Path,
    ) -> Result<Self> {
        Ok(Self {
            working: WorkingMemory::new(working_limit),
            project: ProjectMemory::new(project_db)?,
            user: UserMemory::new(user_db)?,
        })
    }

    /// 写入：按重要性路由到不同层
    pub fn remember_smart(&mut self, entry: MemoryEntry) -> Result<Uuid> {
        match entry.importance {
            crate::common::types::Importance::Low => {
                self.working.remember(entry)
            }
            _ => {
                // Normal 及以上同时写工作记忆和项目记忆
                let entry_clone = entry.clone();
                let id = self.working.remember(entry)?;
                let _ = self.project.remember(entry_clone);
                Ok(id)
            }
        }
    }

    /// 检索：逐层回退
    pub fn recall_smart(&self, query: &str, limit: u8) -> Result<Vec<MemoryEntry>> {
        // 先查工作记忆
        let mut results = self.working.recall(query, RecallScope::Working, limit)?;
        if results.len() >= limit as usize {
            return Ok(results);
        }

        // 工作记忆不够，查项目记忆
        let remaining = limit - results.len() as u8;
        let project_results = self.project.recall(query, RecallScope::Project, remaining)?;
        results.extend(project_results);

        if results.len() >= limit as usize {
            return Ok(results);
        }

        // 还不够，查用户记忆
        let remaining = limit - results.len() as u8;
        let user_results = self.user.recall(query, RecallScope::User, remaining)?;
        results.extend(user_results);

        Ok(results)
    }
}
```

```rust
use crate::common::types::ActionRecord;
use super::reference::ContextRef;

/// 上下文压缩器
pub struct ContextCompressor;

impl ContextCompressor {
    /// L1 结构压缩：工具调用日志 → 摘要行
    pub fn compress_action_log(logs: &[ActionRecord]) -> String {
        if logs.is_empty() {
            return "No actions taken.".into();
        }

        let lines: Vec<String> = logs.iter().map(|a| {
            let status = if a.success { "✓" } else { "✗" };
            format!("{} {} {} — {}", status, a.timestamp.format("%H:%M:%S"), a.tool, &a.summary[..a.summary.len().min(100)])
        }).collect();

        lines.join("\n")
    }

    /// L1 结构压缩：文件内容 → 路径 + 统计信息
    pub fn compress_file_content(path: &str, content: &str) -> (String, ContextRef) {
        let lines = content.lines().count();
        let bytes = content.len();
        let first_lines: Vec<&str> = content.lines().take(3).collect();
        let preview = first_lines.join("\n");

        let summary = format!("文件 {} ({}行, {}字节)", path, lines, bytes);
        let storage_key = format!("file:{}", path);

        let token_cost = (bytes / 4) as u32; // 粗略估算 token 数
        let ctx_ref = ContextRef::new(summary.clone(), storage_key, token_cost);

        (preview, ctx_ref)
    }

    /// L1 结构压缩：错误堆栈 → 错误类型 + 第一行
    pub fn compress_error(error: &str) -> String {
        let first_line = error.lines().next().unwrap_or("unknown error");
        format!("错误: {}", first_line)
    }

    /// L2 语义压缩（v1.0 简化版本 — 规则驱动摘要）
    /// v1.1 引入 LLM 驱动的语义压缩
    pub fn summarize_conversation(messages: &[String], max_summary_len: usize) -> String {
        if messages.len() <= 2 {
            return messages.join("\n");
        }

        // 规则驱动：保留首尾，中间截断
        let mut parts = vec![];
        parts.push(messages.first().cloned().unwrap_or_default());
        parts.push(format!("... ({} 轮对话省略) ...", messages.len().saturating_sub(2)));
        parts.push(messages.last().cloned().unwrap_or_default());

        let summary = parts.join("\n");
        if summary.len() > max_summary_len {
            format!("{}...", &summary[..max_summary_len])
        } else {
            summary
        }
    }

    /// 估算 token 用量
    pub fn estimate_tokens(text: &str) -> u32 {
        // 粗略估算：4 字符 ≈ 1 token
        (text.chars().count() as u32).div_ceil(4)
    }

    /// 检查是否需要压缩
    pub fn needs_compression(current_tokens: u32, max_tokens: u32) -> bool {
        current_tokens > (max_tokens as f64 * 0.8) as u32
    }
}
```

```rust
impl ProfileRegistry {
    pub fn new() -> Self {
        Self {

    /// 从目录加载所有 Profile
    pub fn load_profiles(&mut self, profiles_dir: &Path) -> Result<()> {
        if !profiles_dir.exists() {
            return Ok(());
        }

        for entry in std::fs::read_dir(profiles_dir)
            .map_err(|e| EflowError::Config(format!("read profiles dir: {}", e)))? 
        {
            let entry = entry.map_err(|e| EflowError::Config(format!("read entry: {}", e)))?;
            let path = entry.path();
            if path.extension().map_or(false, |ext| ext == "yaml" || ext == "yml") {
                let profile = self.load_profile_file(&path)?;
                self.profiles.insert(profile.name.clone(), profile);
            }
        }
        Ok(())
    }

    fn load_profile_file(&self, path: &Path) -> Result<Profile> {
        let content = std::fs::read_to_string(path)
            .map_err(|e| EflowError::Config(format!("read {}: {}", path.display(), e)))?;

        // 校验 checksum（v1.0: 简单校验；v2.0: 数字签名）
        let checksum = format!("{:x}", Sha256::digest(content.as_bytes()));
        // checksum 记录在日志中，用于防篡改检测

        let profile: Profile = serde_yaml::from_str(&content)
            .map_err(|e| EflowError::Config(format!("parse {}: {}", path.display(), e)))?;

        tracing::info!("Loaded profile '{}' (checksum: {})", profile.name, &checksum[..8]);
        Ok(profile)
    }

    /// 从目录加载所有 Skill
    pub fn load_skills(&mut self, skills_dir: &Path) -> Result<()> {
        if !skills_dir.exists() {
            return Ok(());
        }

        for entry in std::fs::read_dir(skills_dir)
            .map_err(|e| EflowError::Config(format!("read skills dir: {}", e)))? 
        {
            let entry = entry.map_err(|e| EflowError::Config(format!("read entry: {}", e)))?;
            let path = entry.path();
            if path.extension().map_or(false, |ext| ext == "yaml" || ext == "yml") {

            }
        }

        // (v1.0: 基础校验，v2.0: 沙箱隔离)

        self.skills.insert(skill.name.clone(), skill);
        Ok(())
    }

    /// 获取 Profile

            profile.skills.iter()
                .filter_map(|name| self.skills.get(name))
                .collect()
        } else {
            vec![]
        }
    }

    /// 构建完整的 System Prompt（Profile + 所有 Skill 模板合并）
    pub fn build_system_prompt(&self, profile_name: &str) -> Result<String> {
        let profile = self.get_profile(profile_name)

        let mut prompt = profile.system_prompt.clone();

        for skill in self.get_profile_skills(profile_name) {
            prompt.push_str(&format!("\n\n## Skill: {}\n{}", skill.name, skill.prompt_template));
        }

        Ok(prompt)
    }

    /// 列出所有可用 Profile 名称
    pub fn list_profiles(&self) -> Vec<&String> {
        self.profiles.keys().collect()
    }
```

```rust
pub struct ToolDefinition {
    pub name: String,
    pub description: String,
    pub parameters: serde_json::Value, // JSON Schema
    pub risk_level: RiskLevel,
}

/// 工具输出
#[derive(Debug, Clone)]
pub struct ToolOutput {
    pub success: bool,
    pub content: String,
    pub metadata: Option<serde_json::Value>,
}

/// 工具 trait
#[async_trait]
pub trait Tool: Send + Sync {
    fn definition(&self) -> ToolDefinition;
    async fn execute(&self, params: serde_json::Value) -> Result<ToolOutput>;
}

/// 工具注册表
pub struct ToolRegistry {
    tools: HashMap<String, Arc<dyn Tool>>,
}

impl ToolRegistry {
    pub fn new() -> Self {
        Self { tools: HashMap::new() }
    }

    pub fn register(&mut self, tool: Arc<dyn Tool>) {
        let def = tool.definition();
        self.tools.insert(def.name, tool);
    }

    pub fn get(&self, name: &str) -> Option<&Arc<dyn Tool>> {
        self.tools.get(name)
    }

    /// 获取所有工具定义（用于发送给 LLM）
    pub fn definitions(&self) -> Vec<ToolDefinition> {
        self.tools.values().map(|t| t.definition()).collect()
    }

    /// 执行工具
    pub async fn execute(&self, name: &str, params: serde_json::Value) -> Result<ToolOutput> {
        let tool = self.tools.get(name)
            .ok_or_else(|| crate::common::error::EflowError::Tool(
                format!("Tool '{}' not found", name)
            ))?;

        // 风险检查
        let def = tool.definition();
        if def.risk_level >= RiskLevel::L3 {
            return Err(crate::common::error::EflowError::RiskEscalated {
                task_id: "unknown".into(),
                reason: format!("Tool '{}' requires L3+ permission", name),
            });
        }

        tool.execute(params).await
    }
}
```

```rust
use crate::common::error::Result;
use crate::common::types::*;
use crate::infrastructure::llm::{LlmRouter, Message, ChatRequest};
use super::blackboard::Blackboard;

/// Decisioner — 风险评估 + 执行计划生成 + 模型路由
pub struct Decisioner {
    llm: std::sync::Arc<tokio::sync::Mutex<LlmRouter>>,
}

impl Decisioner {
    pub fn new(llm: std::sync::Arc<tokio::sync::Mutex<LlmRouter>>) -> Self {
        Self { llm }
    }

    /// 评估步骤风险并生成执行计划
    pub async fn decide(&self, blackboard: &Blackboard) -> Result<Blackboard> {
        let step = blackboard.current_step.as_ref()
            .expect("Decisioner called without current_step");

        // 规则先行：已知风险等级直接路由，不额外调 LLM
        let risk = blackboard.risk_level;
        let model_tier = match risk {
            RiskLevel::L0 => ModelTier::Light,
            RiskLevel::L1 => ModelTier::Light,
            RiskLevel::L2 => ModelTier::Medium,
            RiskLevel::L3 => ModelTier::Strong,
        };

        // 检查是否需要拆分子步骤（仅 L2+ 需要 LLM 规划）
        let sub_steps = if risk >= RiskLevel::L2 {
            self.plan_sub_steps(step, risk).await?
        } else {
            vec![step.clone()]
        };

        let execution_plan = ExecutionPlan {
            step: PlannedStep {
                order: 0,
                action: step.action.clone(),
                tool: step.tool.clone(),
                params: step.params.clone(),
                depends_on: None,
            },
            model_tier,
            risk_level: risk,
            sub_steps,
        };

        Ok(blackboard.with_execution_plan(execution_plan))
    }

    /// 对复杂步骤调用 LLM 拆分子步骤
    async fn plan_sub_steps(
        &self,
        step: &TaskStep,
        risk: RiskLevel,
    ) -> Result<Vec<TaskStep>> {
        let mut llm = self.llm.lock().await;

        let messages = vec![
            Message::system("你是一个任务规划专家。将以下操作拆分为更小的子步骤。"),
            Message::user(format!(
                "操作: {}\n工具: {}\n参数: {}\n风险等级: {:?}\n\n请给出拆分后的子步骤列表。",
                step.action, step.tool, step.params, risk
            )),
        ];

        let request = ChatRequest::new("", messages)
            .with_cache(0);

        let response = llm.chat(ModelTier::Strong, request).await?;

        // 简化解析：每行一个子步骤
        let sub_steps: Vec<TaskStep> = response.content.lines()
            .filter(|l| !l.trim().is_empty())
            .map(|line| TaskStep {
                action: line.to_string(),
                tool: step.tool.clone(),
                params: serde_json::json!({}),
                expected_output: None,
            })
            .collect();

        if sub_steps.is_empty() {
            Ok(vec![step.clone()])
        } else {
            Ok(sub_steps)
        }
    }
}
```

```rust
use std::sync::Arc;
use chrono::Utc;

use crate::common::error::Result;
use crate::common::types::*;
use crate::infrastructure::llm::{LlmRouter, Message, ChatRequest};
use crate::capability::tools::ToolRegistry;
use super::blackboard::Blackboard;

/// Executor — 步骤执行
pub struct Executor {
    llm: Arc<tokio::sync::Mutex<LlmRouter>>,
    tools: Arc<ToolRegistry>,
}

impl Executor {
    pub fn new(
        llm: Arc<tokio::sync::Mutex<LlmRouter>>,
        tools: Arc<ToolRegistry>,
    ) -> Self {
        Self { llm, tools }
    }

    /// 执行步骤
    pub async fn execute(&self, blackboard: Blackboard) -> Result<Blackboard> {
        let plan = blackboard.execution_plan.as_ref()
            .expect("Executor called without execution_plan");

        let mut bb = blackboard;
        let start = std::time::Instant::now();

        for sub_step in &plan.sub_steps {
            // 检查是否需要 LLM 推理
            let result = if sub_step.tool.is_empty() || sub_step.tool == "llm_reasoning" {
                // 纯 LLM 推理步骤
                self.execute_llm_step(&plan, sub_step).await?
            } else {
                // 工具执行步骤
                self.execute_tool_step(sub_step).await?
            };

            let duration_ms = start.elapsed().as_millis() as u64;

            let record = ActionRecord {
                timestamp: Utc::now(),
                action: sub_step.action.clone(),
                tool: sub_step.tool.clone(),
                success: result.success,
                summary: result.content.chars().take(200).collect(),
            };

            bb = bb.with_action(record);

            if !result.success {
                // 单步失败不中断整个管道，由 Feedbacker 判定
                tracing::warn!("Step action '{}' returned failure", sub_step.action);
            }
        }

        Ok(bb)
    }

    /// 纯 LLM 推理
    async fn execute_llm_step(
        &self,
        plan: &ExecutionPlan,
        step: &TaskStep,
    ) -> Result<ActionResult> {
        let mut llm = self.llm.lock().await;

        let messages = vec![
            Message::user(format!(
                "任务: {}\n\n请完成任务并提供结果。",
                step.action
            )),
        ];

        let tool_defs = self.tools.definitions();
        let request = ChatRequest::new("", messages)
            .with_tools(tool_defs)
            .with_cache(0);

        let response = llm.chat(plan.model_tier, request).await?;

        Ok(ActionResult {
            success: true,
            output: response.content,
            tool_calls: vec![],
            duration_ms: 0,
        })
    }

    /// 工具执行
    async fn execute_tool_step(&self, step: &TaskStep) -> Result<ActionResult> {
        let start = std::time::Instant::now();
        let output = self.tools.execute(&step.tool, step.params.clone()).await;
        let duration_ms = start.elapsed().as_millis() as u64;

        match output {
            Ok(tool_output) => {
                Ok(ActionResult {
                    success: tool_output.success,
                    output: tool_output.content,
                    tool_calls: vec![ToolCallSummary {
                        tool_name: step.tool.clone(),
                        success: tool_output.success,
                        duration_ms,
                        summary: tool_output.content.chars().take(100).collect(),
                    }],
                    duration_ms,
                })
            }
            Err(e) => {
                Ok(ActionResult {
                    success: false,
                    output: format!("工具执行失败: {}", e),
                    tool_calls: vec![],
                    duration_ms,
                })
            }
        }
    }
}
```

```rust
use std::sync::Arc;
use chrono::Utc;

use crate::common::error::Result;
use crate::common::types::*;
use crate::infrastructure::llm::{LlmRouter, Message, ChatRequest};
use super::blackboard::Blackboard;

/// Feedbacker — 质量评估 + 反馈回路
pub struct Feedbacker {
    llm: Arc<tokio::sync::Mutex<LlmRouter>>,
}

impl Feedbacker {
    pub fn new(llm: Arc<tokio::sync::Mutex<LlmRouter>>) -> Self {
        Self { llm }
    }

    /// 评估执行结果并返回判决
    pub async fn evaluate(&self, blackboard: Blackboard) -> Result<(Blackboard, QualityVerdict)> {
        let last_action = blackboard.action_log.last();

        // 规则先行：无操作记录 → Pass
        let last = match last_action {
            Some(a) => a,
            None => {
                let record = FeedbackRecord {
                    timestamp: Utc::now(),
                    verdict: QualityVerdict::Pass { summary: "no actions to evaluate".into() },
                    retry_count: blackboard.retry_count,
                };
                return Ok((blackboard.with_feedback(record), QualityVerdict::Pass {
                    summary: "no actions".into(),
                }));
            }
        };

        // 规则：全部成功 + 风险 ≤ L1 → 快速 Pass，不调 LLM
        let all_succeeded = blackboard.action_log.iter().all(|a| a.success);
        if all_succeeded && blackboard.risk_level <= RiskLevel::L1 {
            let summary = format!(
                "完成: {} 个操作全部成功",
                blackboard.action_log.len()
            );
            let record = FeedbackRecord {
                timestamp: Utc::now(),
                verdict: QualityVerdict::Pass { summary: summary.clone() },
                retry_count: blackboard.retry_count,
            };
            return Ok((blackboard.with_feedback(record), QualityVerdict::Pass { summary }));
        }

        // LLM 评估（仅复杂情况）
        let verdict = self.llm_evaluate(&blackboard, last).await?;
        let record = FeedbackRecord {
            timestamp: Utc::now(),
            verdict: verdict.clone(),
            retry_count: blackboard.retry_count,
        };

        Ok((blackboard.with_feedback(record), verdict))
    }

    /// 调用 LLM 进行质量评估
    async fn llm_evaluate(
        &self,
        blackboard: &Blackboard,
        last_action: &ActionRecord,
    ) -> Result<QualityVerdict> {
        let mut llm = self.llm.lock().await;

        let operation_summary: String = blackboard.action_log.iter()
            .map(|a| {
                let status = if a.success { "✓" } else { "✗" };
                format!("{} {}: {}", status, a.tool, &a.summary[..a.summary.len().min(80)])
            })
            .collect::<Vec<_>>()
            .join("\n");

        let messages = vec![
            Message::system(
                "你是一个质量评估专家。判断任务执行结果是否达标。\n\
                 回复格式:\n\
                 - PASS: <摘要>  (如果结果达标)\n\
                 - REWORK: <原因> | <建议>  (如果需要重做)\n\
                 - ESCALATE: <原因>  (如果需要升级)"
            ),
            Message::user(format!(
                "任务: {}\n风险等级: {:?}\n重试次数: {}\n\n执行记录:\n{}\n\n请评估。",
                &blackboard.task.description[..blackboard.task.description.len().min(200)],
                blackboard.risk_level,
                blackboard.retry_count,
                operation_summary,
            )),
        ];

        let request = ChatRequest::new("", messages);
        let response = llm.chat(ModelTier::Medium, request).await?;

        // 解析 LLM 输出
        let content = response.content.trim();
        if content.starts_with("PASS:") || content.starts_with("PASS：") {
            Ok(QualityVerdict::Pass {
                summary: content[5..].trim().to_string(),
            })
        } else if content.starts_with("REWORK:") || content.starts_with("REWORK：") {
            let body = &content[7..].trim();
            let parts: Vec<&str> = body.splitn(2, '|').collect();
            let reason = parts.first().unwrap_or(&"未知原因").trim().to_string();
            let suggestion = parts.get(1).unwrap_or(&"请重试").trim().to_string();
            Ok(QualityVerdict::Rework { reason, suggestion })
        } else if content.starts_with("ESCALATE:") || content.starts_with("ESCALATE：") {
            Ok(QualityVerdict::Escalate {
                reason: content[9..].trim().to_string(),
                new_risk: RiskLevel::L3,
            })
        } else {
            // 无法解析 → 默认 Pass
            Ok(QualityVerdict::Pass {
                summary: content.chars().take(100).collect(),
            })
        }
    }
}
```

```rust
use uuid::Uuid;

use crate::common::types::*;
use super::blackboard::Blackboard;
use super::decisioner::Decisioner;
use super::executor::Executor;
use super::feedbacker::Feedbacker;

/// Subagent — 实际执行任务的工人
pub struct Subagent {
    pub id: Uuid,
    pub name: String,
    pub role: Role,
    pub capabilities: Vec<Capability>,
    pub permission: PermissionSet,
}

impl Subagent {
    pub fn new(name: String, role: Role, capabilities: Vec<Capability>) -> Self {
        Self {
            id: Uuid::new_v4(),
            name,
            role,
            capabilities,
            permission: PermissionSet::default(),
        }
    }

    /// 执行单个步骤的完整管线段：D → E → F（含反馈回路）
    pub async fn execute_step(
        &self,
        blackboard: Blackboard,
        decisioner: &Decisioner,
        executor: &Executor,
        feedbacker: &Feedbacker,
    ) -> crate::common::error::Result<Blackboard> {
        let mut bb = blackboard;
        let max_retries = 3;

        loop {
            // Decisioner: 评估 + 规划
            bb = decisioner.decide(&bb).await?;

            // Executor: 执行
            bb = executor.execute(bb).await?;

            // Feedbacker: 评估 + 判决
            let (new_bb, verdict) = feedbacker.evaluate(bb).await?;
            bb = new_bb;

            match verdict {
                QualityVerdict::Pass { .. } => {
                    // 步骤完成
                    return Ok(bb);
                }
                QualityVerdict::Rework { reason: _, suggestion } => {
                    if bb.retry_count >= max_retries {
                        // 超过最大重试，强制升级
                        tracing::warn!(
                            "Step exceeded max retries ({}), escalating", max_retries
                        );
                        bb.risk_level = RiskLevel::L3;
                        return Ok(bb);
                    }
                    // 带着建议重试
                    tracing::info!("Rework needed: {}. Retry {}/{}", suggestion, bb.retry_count + 1, max_retries);
                    bb = bb.increment_retry();
                    // 在当前步骤的 action 上追加修正指令
                    if let Some(ref mut step) = bb.current_step {
                        step.action = format!("{} (修正: {})", step.action, suggestion);
                    }
                    continue;
                }
                QualityVerdict::Escalate { reason, new_risk } => {
                    tracing::warn!("Escalated to {:?}: {}", new_risk, reason);
                    bb.risk_level = new_risk;
                    return Ok(bb);
                }
            }
        }
    }
}
```

```rust
use std::sync::Arc;
use uuid::Uuid;

use crate::common::error::Result;
use crate::common::types::*;

use crate::infrastructure::llm::{LlmRouter, Message, ChatRequest};
use crate::capability::blackboard::Blackboard;
use crate::capability::decisioner::Decisioner;
use crate::capability::executor::Executor;
use crate::capability::feedbacker::Feedbacker;
use crate::capability::subagent::Subagent;
use crate::capability::tools::ToolRegistry;

/// Orchestrator — 任务分解 + Subagent 调度 + 结果聚合
pub struct Orchestrator {
    llm: Arc<tokio::sync::Mutex<LlmRouter>>,
    tools: Arc<ToolRegistry>,
    events: EventChannel,
    active_agent: Option<Subagent>,
}

impl Orchestrator {
    pub fn new(
        llm: Arc<tokio::sync::Mutex<LlmRouter>>,
        tools: Arc<ToolRegistry>,
        events: EventChannel,
    ) -> Self {
        Self {
            llm: llm.clone(),
            tools,
            events,
            active_agent: None,
        }
    }

    /// 确保有一个可用的 Subagent
    fn ensure_agent(&mut self) -> &Subagent {
        if self.active_agent.is_none() {
            self.active_agent = Some(Subagent::new(
                "default".into(),
                Role::Generalist,
                vec![Capability::ReadFile, Capability::WriteFile, Capability::LlmReasoning],
            ));
        }
        self.active_agent.as_ref().unwrap()
    }

    /// 执行任务：TaskSpec → 分解 → 管线段执行 → 聚合结果
    pub async fn execute(&mut self, task: TaskSpec) -> Result<String> {
        let task_id = task.id;

        self.events.publish(Event::TaskStarted {
            task_id,
            description: task.description.clone(),
        });

        // 1. 分解任务为步骤（LLM 驱动）
        let plan = self.decompose(&task).await?;

        // 2. 创建 Blackboard
        let bb = Blackboard::new(task).with_plan(plan);

        // 3. 构建管线段组件
        let decisioner = Decisioner::new(self.llm.clone());
        let executor = Executor::new(self.llm.clone(), self.tools.clone());
        let feedbacker = Feedbacker::new(self.llm.clone());

        // 4. 逐步执行
        let agent = self.ensure_agent();
        let mut bb = bb;
        let total_steps = bb.plan.as_ref().map(|p| p.steps.len()).unwrap_or(0);

        for (i, planned_step) in bb.plan.as_ref().unwrap().steps.iter().enumerate() {
            let step = TaskStep {
                action: planned_step.action.clone(),
                tool: planned_step.tool.clone(),
                params: planned_step.params.clone(),
                expected_output: None,
            };
            bb = bb.with_step(step);

            self.events.publish(Event::TaskStarted {
                task_id,
                description: format!("步骤 {}/{}: {}", i + 1, total_steps, planned_step.action),
            });

            bb = agent.execute_step(bb, &decisioner, &executor, &feedbacker).await?;
        }

        // 5. 聚合结果
        let summary = bb.summarize();

        self.events.publish(Event::TaskCompleted {
            task_id,
            summary: summary.clone(),
        });

        Ok(summary)
    }

    /// LLM 驱动的任务分解
    async fn decompose(&self, task: &TaskSpec) -> Result<TaskPlan> {
        let mut llm = self.llm.lock().await;

        // 简单任务：规则分解
        if task.risk_level <= RiskLevel::L1 && task.description.len() < 100 {
            return Ok(TaskPlan {
                task_id: task.id,
                steps: vec![PlannedStep {
                    order: 0,
                    action: task.description.clone(),
                    tool: "llm_reasoning".into(),
                    params: serde_json::json!({}),
                    depends_on: None,
                }],
                estimated_steps: 1,
                risk_level: task.risk_level,
            });
        }

        // 复杂任务：LLM 分解
        let tool_defs = self.tools.definitions();
        let tools_desc: String = tool_defs.iter()
            .map(|t| format!("- {}: {}", t.name, t.description))
            .collect();

        let messages = vec![
            Message::system(format!(
                "你是一个任务规划专家。将用户任务分解为可执行的步骤序列。\n\
                 可用工具:\n{}\n\
                 输出格式：每行一个步骤，格式为 '工具名: 操作描述'",
                tools_desc
            )),
            Message::user(format!("请分解以下任务:\n{}", task.description)),
        ];

        let request = ChatRequest::new("", messages);
        let response = llm.chat(ModelTier::Strong, request).await?;

        let steps: Vec<PlannedStep> = response.content.lines()
            .filter(|l| !l.trim().is_empty())
            .enumerate()
            .map(|(i, line)| {
                let parts: Vec<&str> = line.splitn(2, ':').collect();
                let tool = parts.first().unwrap_or(&"llm_reasoning").trim();
                let action = parts.get(1).unwrap_or(&"执行操作").trim();

                PlannedStep {
                    order: i as u8,
                    action: action.to_string(),
                    tool: tool.to_string(),
                    params: serde_json::json!({}),
                    depends_on: if i > 0 { Some(i as u8 - 1) } else { None },
                }
            })
            .collect();

        let estimated_steps = steps.len() as u8;
        Ok(TaskPlan {
            task_id: task.id,
            steps,
            estimated_steps,
            risk_level: task.risk_level,
        })
    }
}
```

```rust
/// Concierge — 零阻塞对话入口
pub struct Concierge {
    events: EventChannel,
    memory: Arc<tokio::sync::Mutex<CompositeMemory>>,
    profiles: Arc<tokio::sync::RwLock<ProfileRegistry>>,
    orchestrator: Arc<tokio::sync::Mutex<Orchestrator>>,
    active_profile: String,
}

impl Concierge {
    pub fn new(
        events: EventChannel,
        memory: Arc<tokio::sync::Mutex<CompositeMemory>>,
        profiles: Arc<tokio::sync::RwLock<ProfileRegistry>>,
        orchestrator: Arc<tokio::sync::Mutex<Orchestrator>>,
        default_profile: String,
    ) -> Self {
        Self {
            events,
            memory,
            profiles,
            orchestrator,
            active_profile: default_profile,
        }
    }

    /// 处理用户输入 — 永不阻塞
    pub async fn handle_input(
        &self,
        input: String,
    ) -> String {
        // 1. 意图识别（规则驱动，不调 LLM）
        let intent = self.classify_intent(&input);

        match intent {
            Intent::Chat { content } => {
                format!("收到: {}", content)
            }
            Intent::TaskDispatch { spec } => {
                let task_id = spec.id;
                // 异步派发任务，不等待结果
                let orch = self.orchestrator.clone();
                let events = self.events.clone();
                tokio::spawn(async move {
                    let mut orch = orch.lock().await;
                    match orch.execute(spec).await {
                        Ok(summary) => {
                            events.publish(Event::TaskCompleted {
                                task_id,
                                summary,
                            });
                        }
                        Err(e) => {
                            events.publish(Event::TaskFailed {
                                task_id,
                                error: e.to_string(),
                            });
                        }
                    }
                });
                format!("任务已派发 [{}]，异步执行中...", task_id.to_string().chars().take(8).collect::<String>())
            }
            Intent::TaskInterrupt { task_id } => {
                format!("中断任务 {}", task_id)
            }
            Intent::TaskCancel { task_id } => {
                format!("取消任务 {}", task_id)
            }
            Intent::SkillQuery { keyword } => {
                self.list_skills(&keyword)
            }
            Intent::ProfileSwitch { industry } => {
                format!("切换到 Profile: {}", industry)
            }
        }
    }

    /// 规则驱动的意图分类
    fn classify_intent(&self, input: &str) -> Intent {
        let input_lower = input.to_lowercase();

        // 简单规则：检测关键词
        if input_lower.contains("切换") && input_lower.contains("profile") {
            // 提取 profile 名称
            let parts: Vec<&str> = input.split_whitespace().collect();
            let name = parts.last().unwrap_or(&"developer");
            return Intent::ProfileSwitch { industry: name.to_string() };
        }
        if input_lower.contains("取消") && input_lower.contains("任务") {
            return Intent::TaskCancel { task_id: Uuid::new_v4() };
        }
        if input_lower.contains("中断") {
            return Intent::TaskInterrupt { task_id: Uuid::new_v4() };
        }
        if input_lower.contains("skill") || input_lower.contains("技能") {
            return Intent::SkillQuery { keyword: input.to_string() };
        }

        // 默认：任务派发
        let spec = TaskSpec::new(
            input.to_string(),
            RiskLevel::L0,
        );
        Intent::TaskDispatch { spec }
    }

    fn list_skills(&self, _keyword: &str) -> String {
        // 简化：列出所有 profile
        let profiles = &self.profiles;
        // Cannot block_on in async context — simplified
        "查询 Skills... (v1.0 简化)".into()
    }
}
```

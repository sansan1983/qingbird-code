    61	/// 列出目录内容
    62	pub struct ListDirTool;
    63	
    64	#[async_trait]
    65	impl Tool for ListDirTool {
    66	    fn definition(&self) -> ToolDefinition {
    67	        ToolDefinition {
    68	            name: "list_dir".into(),
    69	            description: t!("tool_desc_list_dir").into(),
    70	            parameters: serde_json::json!({
    71	                "type": "object",
    72	                "properties": {
    73	                    "path": {"type": "string", "description": "目录路径"},
    74	                    "recursive": {"type": "boolean", "description": "是否递归列出子目录"}
    75	                },
    76	                "required": ["path"]
    77	            }),
    78	            risk_level: RiskLevel::L0,
    79	        }
    80	    }
    81	
    82	    async fn execute(&self, params: serde_json::Value) -> Result<ToolOutput> {
    83	        let path = params["path"].as_str().ok_or_else(|| {
    84	            EflowError::Tool(t!("err_tool_missing_param", name = "path").to_string())
    85	        })?;
    86	        let recursive = params["recursive"].as_bool().unwrap_or(false);
    87	
    88	        let root = Path::new(path);
    89	        if !root.exists() {
    90	            return Err(EflowError::Tool(
    91	                t!("err_tool_invalid_path", path = path).to_string(),
    92	            ));
    93	        }
    94	        if !root.is_dir() {
    95	            return Err(EflowError::Tool(
    96	                t!("err_tool_not_a_dir", path = path).to_string(),
    97	            ));
    98	        }
    99	
   100	        let mut entries: Vec<String> = Vec::new();
   101	
   102	        if recursive {
   103	            for entry in WalkDir::new(root)
   104	                .follow_links(false)
   105	                .into_iter()
   106	                .filter_map(std::result::Result::ok)
   107	            {
   108	                let entry_path = entry.path();
   109	                let display = entry_path.display();
   110	                if entry.file_type().is_dir() {
   111	                    entries.push(format!("[DIR]  {}/", display));
   112	                } else if entry.file_type().is_file() {
   113	                    let len = entry.metadata().map(|m| m.len()).unwrap_or(0);
   114	                    entries.push(format!("[FILE] {}  {} bytes", display, len));
   115	                }
   116	            }
   117	        } else {
   118	            let mut read_dir = tokio::fs::read_dir(root).await.map_err(|e| {
   119	                EflowError::Tool(
   120	                    t!("err_tool_read_dir", path = path, msg = e.to_string()).to_string(),
   121	                )
   122	            })?;
   123	
   124	            let mut child_entries: Vec<(String, u64, bool)> = Vec::new();
   125	            while let Some(child) = read_dir.next_entry().await.map_err(|e| {
   126	                EflowError::Tool(
   127	                    t!("err_tool_read_dir", path = path, msg = e.to_string()).to_string(),
   128	                )
   129	            })? {
   130	                let file_type = child.file_type().await.unwrap_or_default();
   131	                let name = child.file_name().to_string_lossy().to_string();
   132	                let len = child.metadata().map(|m| m.len()).unwrap_or(0);
   133	                child_entries.push((name, len, file_type.is_dir()));
   134	            }
   135	
   136	            // 目录在前，文件在后，各自按名称排序
   137	            child_entries.sort_by(|a, b| a.0.cmp(&b.0));
   138	            for (name, len, is_dir) in child_entries {
   139	                if is_dir {
   140	                    entries.push(format!("[DIR]  {}/", name));
   141	                } else {
   142	                    entries.push(format!("[FILE] {}  {} bytes", name, len));
   143	                }
   144	            }
   145	        }
   146	
   147	        let count = entries.len();
   148	        let body = t!("status_list_dir_header", path = path, count = count).to_string()
   149	            + "\n\n"
   150	            + &entries.join("\n");
   151	
   152	        Ok(ToolOutput {
   153	            success: true,
   154	            content: body,
   155	            metadata: Some(serde_json::json!({"entries": count})),
   156	        })
   157	    }
   158	}

   160	/// 按 glob 模式匹配文件
   161	pub struct GlobTool;
   162	
   163	#[async_trait]
   164	impl Tool for GlobTool {
   165	    fn definition(&self) -> ToolDefinition {
   166	        ToolDefinition {
   167	            name: "glob".into(),
   168	            description: t!("tool_desc_glob").into(),
   169	            parameters: serde_json::json!({
   170	                "type": "object",
   171	                "properties": {
   172	                    "pattern": {"type": "string", "description": "glob 模式，如 **/*.rs"},
   173	                    "root": {"type": "string", "description": "搜索根目录，默认 '.'"}
   174	                },
   175	                "required": ["pattern"]
   176	            }),
   177	            risk_level: RiskLevel::L0,
   178	        }
   179	    }
   180	
   181	    async fn execute(&self, params: serde_json::Value) -> Result<ToolOutput> {
   182	        let pattern = params["pattern"].as_str().ok_or_else(|| {
   183	            EflowError::Tool(t!("err_tool_missing_param", name = "pattern").to_string())
   184	        })?;
   185	        let root = params["root"].as_str().unwrap_or(".");
   186	
   187	        let root_path = Path::new(root);
   188	        if !root_path.exists() {
   189	            return Err(EflowError::Tool(
   190	                t!("err_tool_invalid_path", path = root).to_string(),
   191	            ));
   192	        }
   193	
   194	        let mut results: Vec<String> = Vec::new();
   195	        let max_results = 200;
   196	
   197	        for entry in WalkDir::new(root_path)
   198	            .follow_links(false)
   199	            .into_iter()
   200	            .filter_map(std::result::Result::ok)
   201	        {
   202	            if !entry.file_type().is_file() {
   203	                continue;
   204	            }
   205	
   206	            let relative = entry
   207	                .path()
   208	                .strip_prefix(root_path)
   209	                .unwrap_or(entry.path())
   210	                .to_string_lossy()
   211	                .replace('\\', "/");
   212	
   213	            if glob_match(pattern, &relative) {
   214	                results.push(entry.path().display().to_string());
   215	                if results.len() >= max_results {
   216	                    break;
   217	                }
   218	            }
   219	        }
   220	
   221	        let count = results.len();
   222	        let truncated = count >= max_results;
   223	        let body = if results.is_empty() {
   224	            t!("status_no_match", pattern = pattern).to_string()
   225	        } else {
   226	            let header = if truncated {
   227	                t!("status_glob_truncated", count = count, pattern = pattern)
   228	            } else {
   229	                t!("status_glob_header", count = count, pattern = pattern)
   230	            };
   231	            header.to_string() + "\n\n" + &results.join("\n")
   232	        };
   233	
   234	        Ok(ToolOutput {
   235	            success: true,
   236	            content: body,
   237	            metadata: Some(serde_json::json!({
   238	                "matches": count,
   239	                "truncated": truncated,
   240	            })),
   241	        })
   242	    }
   243	}

   245	/// 简易 glob 匹配：将 glob 模式转为正则表达式
   246	///
   247	/// 支持:
   248	/// - `*` — 匹配单层路径内的任意字符（不含 /）
   249	/// - `**` — 匹配跨路径的任意字符
   250	/// - `?` — 匹配单字符（不含 /）
   251	fn glob_match(pattern: &str, path: &str) -> bool {
   252	    let path = path.replace('\\', "/");
   253	
   254	    // 手工构建正则，避免 regex_lite::escape + 替换的顺序问题
   255	    let mut re_str = String::from("^");
   256	    let mut chars = pattern.chars().peekable();
   257	
   258	    while let Some(c) = chars.next() {
   259	        match c {
   260	            '*' if chars.peek() == Some(&'*') => {
   261	                re_str.push_str(".*");
   262	                chars.next(); // 消费第二个 *
   263	            }
   264	            '*' => re_str.push_str("[^/]*"),
   265	            '?' => re_str.push_str("[^/]"),
   266	            '.' => re_str.push_str("\\."),
   267	            // 转义其他正则特殊字符
   268	            '+' | '^' | '$' | '(' | ')' | '[' | ']' | '{' | '}' | '|' | '\\' => {
   269	                re_str.push('\\');
   270	                re_str.push(c);
   271	            }
   272	            c => re_str.push(c),
   273	        }
   274	    }
   275	    re_str.push('$');
   276	
   277	    if let Ok(re) = regex_lite::Regex::new(&re_str) {
   278	        re.is_match(&path)
   279	    } else {
   280	        false
   281	    }
   282	}

   374

   380	/// HTTP GET 请求工具
   381	pub struct HttpGetTool;
   382	
   383	#[async_trait]
   384	impl Tool for HttpGetTool {
   385	    fn definition(&self) -> ToolDefinition {
   386	        ToolDefinition {
   387	            name: "http_get".into(),
   388	            description: t!("tool_desc_http_get").into(),
   389	            parameters: serde_json::json!({
   390	                "type": "object",
   391	                "properties": {
   392	                    "url": {"type": "string", "description": "目标 URL"},
   393	                    "timeout_secs": {"type": "integer", "description": "超时秒数，默认 30"}
   394	                },
   395	                "required": ["url"]
   396	            }),
   397	            risk_level: RiskLevel::L1,
   398	        }
   399	    }
   400	
   401	    async fn execute(&self, params: serde_json::Value) -> Result<ToolOutput> {
   402	        let url = params["url"].as_str().ok_or_else(|| {
   403	            EflowError::Tool(t!("err_tool_missing_param", name = "url").to_string())
   404	        })?;
   405	
   406	        // 检查 URL 合法性
   407	        if !url.starts_with("http://") && !url.starts_with("https://") {
   408	            return Err(EflowError::Tool(
   409	                t!("err_tool_invalid_url", url = url).to_string(),
   410	            ));
   411	        }
   412	
   413	        let timeout_secs = params["timeout_secs"].as_u64().unwrap_or(30);
   414	        let timeout = std::time::Duration::from_secs(timeout_secs);
   415	
   416	        let client = reqwest::Client::builder()
   417	            .timeout(timeout)
   418	            .build()
   419	            .map_err(|e| {
   420	                EflowError::Tool(
   421	                    t!("err_tool_http_client", msg = e.to_string()).to_string(),
   422	                )
   423	            })?;
   424	
   425	        let response = client.get(url).send().await.map_err(|e| {
   426	            EflowError::Tool(
   427	                t!("err_tool_http_get", url = url, msg = e.to_string()).to_string(),
   428	            )
   429	        })?;
   430	
   431	        let status = response.status().as_u16();
   432	        let content_type = response
   433	            .headers()
   434	            .get(reqwest::header::CONTENT_TYPE)
   435	            .and_then(|v| v.to_str().ok())
   436	            .unwrap_or("")
   437	            .to_string();
   438	
   439	        // 只读取前 1MB 内容防止 OOM
   440	        const MAX_BODY: usize = 1_048_576;
   441	        let body = response.text().await.map_err(|e| {
   442	            EflowError::Tool(
   443	                t!("err_tool_http_body", msg = e.to_string()).to_string(),
   444	            )
   445	        })?;
   446	        let truncated = body.len() > MAX_BODY;
   447	        let display_body = if truncated {
   448	            format!("{}...\n(response truncated at {} bytes)", &body[..MAX_BODY], MAX_BODY)
   449	        } else {
   450	            body
   451	        };
   452	
   453	        let content = format!(
   454	            "HTTP {} ({})\n\n{}",
   455	            status, content_type, display_body
   456	        );
   457	
   458	        Ok(ToolOutput {
   459	            success: status < 500,
   460	            content,
   461	            metadata: Some(serde_json::json!({
   462	                "status_code": status,
   463	                "content_type": content_type,
   464	                "truncated": truncated,
   465	            })),
   466	        })
   467	    }
   468	}

   552	/// 精准替换文件中的文本（需唯一匹配）
   553	pub struct EditFileTool;
   554	
   555	#[async_trait]
   556	impl Tool for EditFileTool {
   557	    fn definition(&self) -> ToolDefinition {
   558	        ToolDefinition {
   559	            name: "edit_file".into(),
   560	            description: t!("tool_desc_edit_file").into(),
   561	            parameters: serde_json::json!({
   562	                "type": "object",
   563	                "properties": {
   564	                    "path": {"type": "string", "description": "文件路径"},
   565	                    "old_string": {"type": "string", "description": "被替换的文本（需唯一匹配）"},
   566	                    "new_string": {"type": "string", "description": "替换后的文本"}
   567	                },
   568	                "required": ["path", "old_string", "new_string"]
   569	            }),
   570	            risk_level: RiskLevel::L1,
   571	        }
   572	    }

   576	            EflowError::Tool(t!("err_tool_missing_param", name = "path").to_string())
   577	        })?;
   578	        let old_string = params["old_string"].as_str().ok_or_else(|| {
   579	            EflowError::Tool(t!("err_tool_missing_param", name = "old_string").to_string())
   580	        })?;
   581	        let new_string = params["new_string"].as_str().ok_or_else(|| {
   582	            EflowError::Tool(t!("err_tool_missing_param", name = "new_string").to_string())
   583	        })?;
   584	
   585	        let file_path = Path::new(path);
   586	        if !file_path.exists() {
   587	            return Err(EflowError::Tool(
   588	                t!("err_tool_invalid_path", path = path).to_string(),
   589	            ));
   590	        }
   591	
   592	        let content = tokio::fs::read_to_string(file_path).await.map_err(|e| {
   593	            EflowError::Tool(
   594	                t!("err_tool_read_file", path = path, msg = e.to_string()).to_string(),
   595	            )
   596	        })?;
   597	
   598	        // 检查匹配次数
   599	        let count = content.matches(old_string).count();
   600	        if count == 0 {
   601	            return Err(EflowError::Tool(
   602	                t!("err_tool_edit_not_found", old = old_string).to_string(),
   603	            ));
   604	        }
   605	        if count > 1 {
   606	            // 显示前 3 处上下文帮助调试
   607	            let mut contexts: Vec<String> = Vec::new();
   608	            for (i, line) in content.lines().enumerate() {
   609	                if line.contains(old_string) {
   610	                    contexts.push(format!("  line {}: {}", i + 1, line));
   611	                    if contexts.len() >= 3 {
   612	                        break;
   613	                    }
   614	                }
   615	            }
   616	            return Err(EflowError::Tool(
   617	                t!(
   618	                    "err_tool_edit_ambiguous",
   619	                    count = count,
   620	                    old = old_string,
   621	                    contexts = contexts.join("\n")
   622	                )
   623	                .to_string(),
   624	            ));
   625	        }
   626	
   627	        // 唯一匹配，安全替换
   628	        let new_content = content.replace(old_string, new_string);
   629	        tokio::fs::write(file_path, new_content).await.map_err(|e| {
   630	            EflowError::Tool(
   631	                t!("err_tool_write_file", path = path, msg = e.to_string()).to_string(),
   632	            )
   633	        })?;
   634	
   635	        let bytes = content.len();
   636	        Ok(ToolOutput {
   637	            success: true,
   638	            content: t!("status_edit_success", path = path).to_string(),
   639	            metadata: Some(serde_json::json!({
   640	                "file_size": bytes,
   641	            })),
   642	        })
   643	    }

   833	///
   834	/// ADR-V0.1.1-002: 输出到 stderr，保持 stdout 干净
   835	pub struct Spinner {
   836	    stop_tx: Option<tokio::sync::oneshot::Sender<()>>,
   837	}
   838	
   839	impl Spinner {
   840	    /// 启动 spinner（自动在 stderr 上开始动画，返回句柄）
   841	    pub fn start(message: &str) -> Self {
   842	        let frames = ["⠋", "⠙", "⠹", "⠸", "⠼", "⠴", "⠦", "⠧", "⠇", "⠏"];
   843	        let (tx, mut rx) = tokio::sync::oneshot::channel::<()>();
   844	        let msg = message.to_string();
   845	
   846	        tokio::spawn(async move {
   847	            let mut i: usize = 0;
   848	            loop {
   849	                tokio::select! {
   850	                    _ = &mut rx => {
   851	                        // 清除 spinner 行
   852	                        use std::io::Write;
   853	                        let _ = write!(std::io::stderr(), "\r\x1b[K");
   854	                        let _ = std::io::stderr().flush();
   855	                        break;
   856	                    }
   857	                    _ = tokio::time::sleep(std::time::Duration::from_millis(120)) => {
   858	                        use std::io::Write;
   859	                        let _ = write!(
   860	                            std::io::stderr(),
   861	                            "\r{} {}",
   862	                            frames[i % frames.len()],
   863	                            msg
   864	                        );
   865	                        let _ = std::io::stderr().flush();
   866	                        i += 1;
   867	                    }
   868	                }
   869	            }
   870	        });
   871	
   872	        Self { stop_tx: Some(tx) }
   873	    }
   874	
   875	    /// 停止 spinner，清除输出行
   876	    pub fn stop(mut self) {
   877	        if let Some(tx) = self.stop_tx.take() {
   878	            let _ = tx.send(());
   879	        }
   880	    }
   881	}
   882	
   883	impl Drop for Spinner {
   884	    fn drop(&mut self) {
   885	        if let Some(tx) = self.stop_tx.take() {
   886	            let _ = tx.send(());
   887	        }
   888	    }
   889	}
